94int TProcuraMelhorativa::EscaladaDoMonte()
95{
96 bool movePrimeiro = (parametro[movePrimeiroEM].valor == 1);
97 TPonto atual = this; // estado atual
98 TPonto melhor = this; // melhor estado para todas as escaladas
99 int procuras = 0;
100 TPonto solucao;
101 NovaSolucao();
102 melhor->custo = atual->custo = Avaliar(); // avaliar o proprio estado
103 // estado atual distinto do melhor
104 atual = solucao = (TProcuraMelhorativa*)Duplicar();
105 while (!melhor->Parar()) {
106 // iniciar uma escalada
107 epocas++;
108 // ponto de partida uma solução aleatória (local no espaço de soluções completas)
109 solucao->NovaSolucao();
110 atual->custo = solucao->Avaliar();
111 DebugInicioEM(++procuras, solucao);
112
113 while (!melhor->Parar()) {
114 // ver de entre os vizinhos, qual o melhor, para avançar um degrau
115 TVector<TPonto> vizinhos;
116 bool alterado = false;
117 solucao->Vizinhanca(vizinhos);
118 if (movePrimeiro) {
119 // move logo que encontre um vizinho melhor
120 // (pode estar a ser ignorados vizinhos melhores)
121 while (vizinhos.Count() > 0) {
122 ((TProcuraMelhorativa*)vizinhos.Last())->Avaliar();
123 if (atual->custo > vizinhos.Last()->custo) {
124 solucao = MelhorAtual(atual, vizinhos, vizinhos.Count() - 1);
125 VerificaMelhor(melhor, atual);
126 // novos vizinhos a partir do estado atual são apagados
127 LibertarVector(vizinhos);
128 alterado = true;
129 }
130 else {
131 delete vizinhos.Last();
132 vizinhos.Pop();
133 }
134 }
135 }
136 else {
137 // analisa todos os vizinhos e avança apenas para o melhor vizinho
138 int melhorValor, melhorIndice;
139 CalcularAvaliacoes(vizinhos, melhorValor, melhorIndice);
140 // trocar caso melhore
141 if (atual->custo > melhorValor) {
142 solucao = MelhorAtual(atual, vizinhos, melhorIndice);
143 VerificaMelhor(melhor, atual);
144 // novos vizinhos a partir do estado atual
145 alterado = true;
146 }
147 // tudo explorado, libertar
148 LibertarVector(vizinhos);
149 }
150 // final da análise da vizinhança, ver se se subiu alguma coisa
151 if (!alterado) {
152 // significa que estamos num óptimo local, no topo do moonte
153 DebugOptimoLocal(solucao);
154 break; // recomeçar nova escalada
155 }
156 else
157 DebugPassoEM(solucao);
158 }
159 }
160 //delete atual;
161 delete solucao;
162
163 return melhor->custo;
164}

254// Algoritmo de procura local: Algoritmos Geneticos
256// retorna a avaliacao do resultado final
257// parametros para a mutacao e cruzamento podem ser dados em variaveis globais
258int TProcuraMelhorativa::AlgoritmoGenetico()
259{
260 int populacao = parametro[populacaoAE].valor;
261 int probablidadeMutacao = parametro[probMutarAE].valor;
262 int distanciaMinima = parametro[distMinimaAG].valor;
263 TVector<TPonto> geracaoAntiga, geracaoNova;
264 TVector<int> id;
265 TPonto melhor = this; // melhor estado para toda a procura
266 NovaSolucao();
267 melhor->Avaliar(); // avaliar o proprio
268 if (parametro[nivelDebug].valor > 1)
269 printf("\nPopulação inicial (%d elementos)", populacao);
270 // construir a geracao inicial
271 for (int i = 0; i < populacao; i++) {
272 geracaoAntiga.Add((TPonto)Duplicar());
273 geracaoAntiga.Last()->NovaSolucao();
274 geracaoAntiga.Last()->Avaliar();
275 VerificaMelhor(melhor, geracaoAntiga.Last());
276 }
277 while (!melhor->Parar()) {
278 // iniciar uma nova geração, com base na antiga
279 epocas++;
280
282 // cruzar conforme o custo
284 int minimo, maximo, total;
285 TVector<int> probabilidade;
286 total = 0;
287 ObterExtremos(geracaoAntiga, minimo, maximo);
288 DebugPassoAG(geracaoAntiga.Count(), minimo, maximo);
289 // calcular a probabilidade de um elemento ser selecionado
290 for (int i = 0; i < geracaoAntiga.Count(); i++) {
291 // cada elemento tem de ter sempre probabilidade de ser escolhido
292 // por isso adiciona-se 1 para que o custo maximo tenha probabilidade minima
293 probabilidade.Add(maximo - geracaoAntiga[i]->custo + 1);
294 total += probabilidade.Last();
295 }
296 // gerar novos elementos. gera-se o doubro para poder seleccionar os melhores
297 while (geracaoNova.Count() < 2 * populacao) {
298 // escolher dois individuos aleatoriamente de acordo com as probabilidades
299 int pai, mae, mutou = 0;
300 Selecao(pai, mae, probabilidade, total);
301 // gerar um novo individuo por cruzamento
302 geracaoNova.Add((TPonto)Duplicar());
303 geracaoNova.Last()->Cruzamento(geracaoAntiga[pai], geracaoAntiga[mae]);
304 // mudar o novo elemento, dependente da probabilidade
305 if (TRand::rand() % 100 < (unsigned)probablidadeMutacao) {
306 mutou = 1;
307 geracaoNova.Last()->Mutar();
308 }
309 // avaliar o valor do elemento
310 geracaoNova.Last()->Avaliar();
311 DebugCruzamentoAG(
312 geracaoAntiga[pai]->custo,
313 geracaoAntiga[mae]->custo,
314 geracaoNova.Last()->custo,
315 mutou);
316 VerificaMelhor(melhor, geracaoNova.Last());
317 }
318
319 // a nova geração já tem elementos, apagar a antiga
320 LibertarVector(geracaoAntiga);
321 // seleccao: os melhores, com base no valor
322 OrdemValor(geracaoNova, id);
323 // remover os elementos que são muito parecidos
324 for (int i = 0; i < populacao && i < geracaoNova.Count(); i++) {
325 // verificar se ha outro elemento muito parecido la
326 if (i > 0 && distanciaMinima > 0) {
327 bool diferente = true;
328 for (int j = 0; j < geracaoAntiga.Count(); j++)
329 if (geracaoAntiga[j]->Distancia(geracaoNova[id[i]]) < distanciaMinima) {
330 diferente = false;
331 break;
332 }
333 if (!diferente)
334 continue;
335 }
336 // este elemento fica, passando para a geração antiga
337 geracaoAntiga.Add(geracaoNova[id[i]]);
338 geracaoNova[id[i]] = NULL;
339 }
340 // todos os novos que não se mantêm, são apagados
341 LibertarVector(geracaoNova);
342 // tudo pronto para uma nova geração
343 }
344 // não há mais gerações, apagar tudo
345 LibertarVector(geracaoAntiga);
346 return melhor->custo;
347}

397 TVector<TPonto> populacao, pais, descendentes;

443TVector<TPonto> TProcuraMelhorativa::SelecionarPaisAE(TVector<TPonto>& populacao) {
444 TVector<TPonto> pais;
445 // selecionar os pais, de acordo com o método escolhido
446 int descendentes = (Parametro(percDescendentesAE) * populacao.Count()) / 100;
447 int pop = populacao.Count();
448 if (descendentes < 1)
449 descendentes = 1;
450 else if (descendentes > pop)
451 descendentes = pop;
452 pais.Count(0);
453 if (Parametro(selecaoAE) == 1) { // roleta
454 // roleta implementada como Stochastic Universal Sampling (SUS)
455 TVector<int> id;
456 TVector<int> custos;
457 TVector<float> probabilidades;
458 int pressao = Parametro(pressaoAE);
459 OrdemValor(populacao, id);
460 for (int i = 0; i < populacao.Count(); i++) {
461 // calcular a probabilidade de um elemento ser selecionado
462 // utilizando o método Ranking Selection
463 probabilidades.Add((2.0f - (float)pressao / 100.0f) / pop +
464 (2.0f * id[i] * ((float)pressao / 100.0f - 1.0f)) / (float)(pop * (pop - 1)));
465 if (i > 0)
466 probabilidades[i] += probabilidades[i - 1];
467 }
468 // escolher os pais (roleta implementado como SUS)
469 float valor = (float)(TRand::rand() % 10000) / (descendentes * 10000.0f);
470 for (int escolhido = 0; pais.Count() < descendentes; escolhido++) {
471 while (pais.Count() < descendentes && valor <= probabilidades[escolhido]) {
472 pais += populacao[escolhido];
473 valor += 1.0f / (float)descendentes;
474 }
475 }
476 }
477 else if (Parametro(selecaoAE) == 3) { // uniforme
478 for (int i = 0; i < descendentes; i++)
479 pais += populacao.Random();
480 }
481 else if (Parametro(selecaoAE) == 2) { // torneio
482 int tamanho = Parametro(tamanhoTorneioAE);
483 for (int i = 0; i < descendentes; i++) {
484 TVector<TPonto> competidores;
485 TVector<int> id;
486 for (int j = 0; j < tamanho; j++)
487 competidores += populacao.Random();
488 // ordenar por custo
489 OrdemValor(competidores, id);
490 // escolher melhor ou segundo melhor
491 if ((TRand::rand() % 100) < Parametro(probMelhorTorneioAE))
492 pais += competidores[id[0]];
493 else
494 pais += competidores[id[1 % tamanho]];
495 }
496 }
497 return pais;
498}

500TVector<TPonto> TProcuraMelhorativa::ReproduzirAE(TVector<TPonto>& pais) {
501 TVector<TPonto> descendentes;
502 pais.RandomOrder();
503 while (pais.Count() > 0) {
504 TPonto pai = pais.Pop();
505
506 if(pais.Count() == 0) {
507 // não há mãe, apenas copiar o pai
508 descendentes += pai->Duplicar();
509 }
510 else {
511 TPonto mae = pais.Pop();
512 if (TRand::rand() % 100 < Parametro(probCruzarAE)) {
513 // gerar um novo individuo por cruzamento
514 TPonto filho = pai->Duplicar();
515 filho->Cruzamento(pai, mae);
516 descendentes += filho;
517 filho = mae->Duplicar();
518 filho->Cruzamento(mae, pai);
519 descendentes += filho;
520 }
521 else {
522 // copia o elemento sem alterações
523 descendentes += pai->Duplicar();
524 descendentes += mae->Duplicar();
525 }
526 }
527 }
528 // mutar e avaliar descendentes
529 for (auto descendente : descendentes) {
530 // mudar o novo elemento, dependente da probabilidade
531 if (TRand::rand() % 100 < Parametro(probMutarAE))
532 descendente->Mutar();
533 // avaliar o valor do elemento
534 descendente->Avaliar();
535 VerificaMelhor(descendente);
536 }
537
538 return descendentes;
539}

541TVector<TPonto> TProcuraMelhorativa::SelecionarSobreviventesAE(TVector<TPonto>& populacao, TVector<TPonto>& descendentes) {
542 TVector<TPonto> elite;
543 int nElite = Parametro(elitismoAE);
544 int melhorDescendente = INT_MAX;
545 if (nElite > 0) {
546 // 1. Copiar os N melhores da população atual
547 TVector<int> id;
548 OrdemValor(populacao, id); // melhor primeiro
549 for (int i = 0; i < nElite && i < populacao.Count(); i++)
550 elite += populacao[id[i]]->Duplicar();
551 // encontrar o melhor descendente
552 melhorDescendente = MelhorCusto(descendentes);
553 }
554
555 if (Parametro(sobrevivenciaAE) == 1) { // idade
556 // remover os mais velhos (os que estão mais tempo na população)
557 for (int i = 0; i < descendentes.Count() && i < populacao.Count(); i++) {
558 delete populacao[i];
559 populacao[i] = NULL;
560 }
561 populacao.Remove(NULL);
562 populacao += descendentes;
563 descendentes.Count(0);
564 }
565 else if (Parametro(sobrevivenciaAE) == 2) { // substituir piores
566 populacao += descendentes;
567 descendentes.Count(0);
568 TVector<int> id;
569 OrdemValor(populacao, id); // ordena por custo (melhor primeiro)
570 TVector<TPonto> novaPopulacao;
571 for (int i = 0; i < Parametro(populacaoAE); i++) {
572 novaPopulacao += populacao[id[i]];
573 populacao[id[i]] = NULL;
574 }
575 LibertarVector(populacao);
576 populacao = novaPopulacao;
577 }
578 else if (Parametro(sobrevivenciaAE) == 3) { // round-robin
579 // cada elemento compete com q outros, os que perdem mais são eliminados
580 int q = Parametro(qRoundRobinAE); // número de competidores
581 TVector<int> perdas;
582 populacao += descendentes;
583 descendentes.Count(0);
584 perdas.Count(populacao.Count());
585 perdas.Reset(0);
586 for (int i = 0; i < populacao.Count(); i++) {
587 TPonto atual = populacao[i];
588 for (int j = 0; j < q; j++) {
589 TPonto oponente;
590 do {
591 oponente = populacao.Random();
592 } while (oponente == atual && populacao.Count() > 1);
593 if (atual->custo > oponente->custo)
594 perdas[i]++;
595 }
596 }
597 // ordenar por número de perdas
598 TVector<int> id;
599 perdas.Sort(&id);
600 TVector<TPonto> novaPopulacao;
601 for (int i = 0; i < Parametro(populacaoAE); i++) {
602 novaPopulacao += populacao[id[i]];
603 populacao[id[i]] = NULL;
604 }
605 LibertarVector(populacao);
606 populacao = novaPopulacao;
607 }
608
609 if (nElite > 0) {
610 // 2. Garantir que os elites estão presentes
611 // (apenas se forem melhores que o melhor descendente)
612 for (auto& e : elite) {
613 if (e->custo < melhorDescendente) {
614 // remover um aleatório para dar lugar ao elite
615 int idx = TRand::rand() % populacao.Count();
616 delete populacao[idx];
617 populacao[idx] = e;
618 }
619 else {
620 delete e; // já não é necessário
621 }
622 }
623 }
624
625 return populacao;
626}
