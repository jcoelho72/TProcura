63int TProcuraMelhorativa::EscaladaDoMonte()
64{
65 bool movePrimeiro = (parametro[movePrimeiroEM].valor == 1);
66 TPonto atual = this; // estado atual
67 TPonto melhor = this; // melhor estado para todas as escaladas
68 int procuras = 0;
69 TPonto solucao;
70 NovaSolucao();

72 // estado atual distinto do melhor
73 atual = solucao = (TProcuraMelhorativa*)Duplicar();
74 while (!melhor->Parar()) {
75 // iniciar uma escalada
76 epocas++;
77 // ponto de partida uma solução aleatória (local no espaço de soluções completas)
78 solucao->NovaSolucao();
79 atual->custo = solucao->Avaliar();
80 DebugInicioEM(++procuras, solucao);
81
82 while (!melhor->Parar()) {
83 // ver de entre os vizinhos, qual o melhor, para avançar um degrau
84 TVector<TPonto> vizinhos;
85 bool alterado = false;
86 solucao->Vizinhanca(vizinhos);
87 if (movePrimeiro) {
88 // move logo que encontre um vizinho melhor
89 // (pode estar a ser ignorados vizinhos melhores)
90 while (vizinhos.Count() > 0) {
91 ((TProcuraMelhorativa*)vizinhos.Last())->Avaliar();
92 if (atual->custo > vizinhos.Last()->custo) {
93 solucao = MelhorAtual(atual, vizinhos, vizinhos.Count() - 1);
94 VerificaMelhor(melhor, atual);
95 // novos vizinhos a partir do estado atual são apagados
96 LibertarVector(vizinhos);
97 alterado = true;
98 }
99 else {
100 delete vizinhos.Last();
101 vizinhos.Pop();
102 }
103 }
104 }
105 else {
106 // analisa todos os vizinhos e avança apenas para o melhor vizinho
107 int melhorValor, melhorIndice;
108 CalcularAvaliacoes(vizinhos, melhorValor, melhorIndice);
109 // trocar caso melhore
110 if (atual->custo > melhorValor) {

112 VerificaMelhor(melhor, atual);
113 // novos vizinhos a partir do estado atual

115 }
116 // tudo explorado, libertar

118 }

208// parametros para a mutacao e cruzamento podem ser dados em variaveis globais
209int TProcuraMelhorativa::AlgoritmoGenetico()
210{
211 int populacao = parametro[populacaoAG].valor;
212 int probablidadeMutacao = parametro[probMutacaoAG].valor;
213 int distanciaMinima = parametro[distMinimaAG].valor;
214 TVector<TPonto> geracaoAntiga, geracaoNova;
215 TVector<int> id;
216 TPonto melhor = this; // melhor estado para toda a procura
217 NovaSolucao();
218 melhor->Avaliar(); // avaliar o proprio
219 if (parametro[nivelDebug].valor > 1)
220 printf("\nPopulação inicial (%d elementos)", populacao);
221 // construir a geracao inicial
222 for (int i = 0; i < populacao; i++) {
223 geracaoAntiga.Add((TPonto)Duplicar());
224 geracaoAntiga.Last()->NovaSolucao();
225 geracaoAntiga.Last()->Avaliar();
226 VerificaMelhor(melhor, geracaoAntiga.Last());
227 }
228 while (!melhor->Parar()) {
229 // iniciar uma nova geração, com base na antiga
230 epocas++;
231
233 // cruzar conforme o custo
235 int minimo, maximo, total;
236 TVector<int> probabilidade;
237 total = 0;
238 ObterExtremos(geracaoAntiga, minimo, maximo);
239 DebugPassoAG(geracaoAntiga.Count(), minimo, maximo);
240 // calcular a probabilidade de um elemento ser selecionado
241 for (int i = 0; i < geracaoAntiga.Count(); i++) {
242 // cada elemento tem de ter sempre probabilidade de ser escolhido
243 // por isso adiciona-se 1 para que o custo maximo tenha probabilidade minima
244 probabilidade.Add(maximo - geracaoAntiga[i]->custo + 1);
245 total += probabilidade.Last();
246 }
247 // gerar novos elementos. gera-se o doubro para poder seleccionar os melhores
248 while (geracaoNova.Count() < 2 * populacao) {
249 // escolher dois individuos aleatoriamente de acordo com as probabilidades
250 int pai, mae, mutou = 0;
251 Selecao(pai, mae, probabilidade, total);
252 // gerar um novo individuo por cruzamento
253 geracaoNova.Add((TPonto)Duplicar());
254 geracaoNova.Last()->Cruzamento(geracaoAntiga[pai], geracaoAntiga[mae]);
255 // mudar o novo elemento, dependente da probabilidade
256 if (TRand::rand() % 100 < (unsigned)probablidadeMutacao) {
257 mutou = 1;
258 geracaoNova.Last()->Mutar();
259 }
260 // avaliar o valor do elemento
261 geracaoNova.Last()->Avaliar();
262 DebugCruzamentoAG(
263 geracaoAntiga[pai]->custo,
264 geracaoAntiga[mae]->custo,
265 geracaoNova.Last()->custo,
266 mutou);
267 VerificaMelhor(melhor, geracaoNova.Last());
268 }
269
270 // a nova geração já tem elementos, apagar a antiga
271 LibertarVector(geracaoAntiga);
272 // seleccao: os melhores, com base no valor
273 OrdemValor(geracaoNova, id);
274 // remover os elementos que são muito parecidos
275 for (int i = 0; i < populacao && i < geracaoNova.Count(); i++) {
276 // verificar se ha outro elemento muito parecido la
277 if (i > 0 && distanciaMinima > 0) {
278 bool diferente = true;
279 for (int j = 0; j < geracaoAntiga.Count(); j++)
280 if (geracaoAntiga[j]->Distancia(geracaoNova[id[i]]) < distanciaMinima) {
281 diferente = false;
282 break;
283 }
284 if (!diferente)
285 continue;
286 }
287 // este elemento fica, passando para a geração antiga
288 geracaoAntiga.Add(geracaoNova[id[i]]);
289 geracaoNova[id[i]] = NULL;
290 }
291 // todos os novos que não se mantêm, são apagados
292 LibertarVector(geracaoNova);
293 // tudo pronto para uma nova geração
294 }
295 // não há mais gerações, apagar tudo
296 LibertarVector(geracaoAntiga);
297 return melhor->custo;
298}
