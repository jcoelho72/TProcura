100// retorna a avaliacao do resultado actual
101int TProcuraMelhorativa::EscaladaDoMonte()
102{
103 bool movePrimeiro = true; //(Parametro(MOVE_PRIMEIRO) == 1);
104 TPonto atual = this; // estado atual
105 TPonto melhor = this; // melhor estado para todas as escaladas
106 int procuras = 0;
107 TPonto solucao;
108 NovaSolucao();
109 melhor->custo = atual->custo = Avaliar(); // avaliar o proprio estado
110 // estado atual distinto do melhor
111 atual = solucao = (TProcuraMelhorativa*)Duplicar();
112 while (!melhor->Parar()) {
113 // iniciar uma escalada
114 epocas++;
115 // ponto de partida uma solução aleatória (local no espaço de soluções completas)
116 solucao->NovaSolucao();
117 atual->custo = solucao->Avaliar();
118 DebugInicioEM(++procuras, solucao);
119
120 while (!melhor->Parar()) {
121 // ver de entre os vizinhos, qual o melhor, para avançar um degrau
122 TVector<TPonto> vizinhos;
123 bool alterado = false;
124 solucao->Vizinhanca(vizinhos);
125 if (movePrimeiro) {
126 // move logo que encontre um vizinho melhor
127 // (pode estar a ser ignorados vizinhos melhores)
128 while (!vizinhos.Empty()) {
129 ((TProcuraMelhorativa*)vizinhos.Last())->Avaliar();
130 if (atual->custo > vizinhos.Last()->custo) {
131 solucao = MelhorAtual(atual, vizinhos, vizinhos.Count() - 1);
132 VerificaMelhor(melhor, atual);
133 // novos vizinhos a partir do estado atual são apagados
134 LibertarVector(vizinhos);
135 alterado = true;
136 }
137 else {
138 delete vizinhos.Last();
139 vizinhos.Pop();
140 }
141 }
142 }
143 else {
144 // analisa todos os vizinhos e avança apenas para o melhor vizinho
145 int melhorValor = 0, melhorIndice = 0;
146 CalcularAvaliacoes(vizinhos, melhorValor, melhorIndice);
147 // trocar caso melhore
148 if (atual->custo > melhorValor) {
149 solucao = MelhorAtual(atual, vizinhos, melhorIndice);
150 VerificaMelhor(melhor, atual);
151 // novos vizinhos a partir do estado atual
152 alterado = true;
153 }
154 // tudo explorado, libertar
155 LibertarVector(vizinhos);
156 }
157 // final da análise da vizinhança, ver se se subiu alguma coisa
158 if (!alterado) {
159 // significa que estamos num óptimo local, no topo do moonte
160 DebugOptimoLocal(solucao);
161 break; // recomeçar nova escalada
162 }
163 else
164 DebugPassoEM(solucao);
165 }
166 }
167 //delete atual;
168 delete solucao;
169
170 return melhor->custo;
171}

248// Algoritmo de procura local: Algoritmos Geneticos
250// retorna a avaliacao do resultado final
251// parametros para a mutacao e cruzamento podem ser dados em variaveis globais
252int TProcuraMelhorativa::AlgoritmoGenetico()
253{
254 int populacao = Parametro(POPULACAO);
255 int probablidadeMutacao = Parametro(PROB_MUTAR);
256 int distanciaMinima = Parametro(DIST_MINIMA);
257 TVector<TPonto> geracaoAntiga, geracaoNova;
258 TVector<int> id;
259 TPonto melhor = this; // melhor estado para toda a procura
260 NovaSolucao();
261 melhor->Avaliar(); // avaliar o proprio
262 Debug(PASSOS, false, "\nPopulação inicial (%d elementos)", populacao);
263 // construir a geracao inicial
264 for (int i = 0; i < populacao; i++) {
265 geracaoAntiga += ((TPonto)Duplicar());
266 geracaoAntiga.Last()->NovaSolucao();
267 geracaoAntiga.Last()->Avaliar();
268 VerificaMelhor(melhor, geracaoAntiga.Last());
269 }
270 while (!melhor->Parar()) {
271 // iniciar uma nova geração, com base na antiga
272 epocas++;
273
275 // cruzar conforme o custo
277 int minimo, maximo, total;
278 TVector<int> probabilidade;
279 total = 0;
280 ObterExtremos(geracaoAntiga, minimo, maximo);
281 DebugPassoAG(geracaoAntiga.Count(), minimo, maximo);
282 // calcular a probabilidade de um elemento ser selecionado
283 for (int i = 0; i < geracaoAntiga.Count(); i++) {
284 // cada elemento tem de ter sempre probabilidade de ser escolhido
285 // por isso adiciona-se 1 para que o custo maximo tenha probabilidade minima
286 probabilidade += (maximo - geracaoAntiga[i]->custo + 1);
287 total += probabilidade.Last();
288 }
289 // gerar novos elementos. gera-se o doubro para poder seleccionar os melhores
290 while (geracaoNova.Count() < 2 * populacao) {
291 // escolher dois individuos aleatoriamente de acordo com as probabilidades
292 int pai, mae, mutou = 0;
293 Selecao(pai, mae, probabilidade, total);
294 // gerar um novo individuo por cruzamento
295 geracaoNova += ((TPonto)Duplicar());
296 geracaoNova.Last()->Cruzamento(geracaoAntiga[pai], geracaoAntiga[mae]);
297 // mudar o novo elemento, dependente da probabilidade
298 if (TRand::rand() % 100 < (unsigned)probablidadeMutacao) {
299 mutou = 1;
300 geracaoNova.Last()->Mutar();
301 }
302 // avaliar o valor do elemento
303 geracaoNova.Last()->Avaliar();
304 DebugCruzamentoAG(
305 geracaoAntiga[pai]->custo,
306 geracaoAntiga[mae]->custo,
307 geracaoNova.Last()->custo,
308 mutou);
309 VerificaMelhor(melhor, geracaoNova.Last());
310 }
311
312 // a nova geração já tem elementos, apagar a antiga
313 LibertarVector(geracaoAntiga);
314 // seleccao: os melhores, com base no valor
315 OrdemValor(geracaoNova, id);
316 // remover os elementos que são muito parecidos
317 for (int i = 0; i < populacao && i < geracaoNova.Count(); i++) {
318 // verificar se ha outro elemento muito parecido la
319 if (i > 0 && distanciaMinima > 0) {
320 bool diferente = true;
321 for (int j = 0; j < geracaoAntiga.Count(); j++)
322 if (geracaoAntiga[j]->Distancia(geracaoNova[id[i]]) < distanciaMinima) {
323 diferente = false;
324 break;
325 }
326 if (!diferente)
327 continue;
328 }
329 // este elemento fica, passando para a geração antiga
330 geracaoAntiga += geracaoNova[id[i]];
331 geracaoNova[id[i]] = NULL;
332 }
333 // todos os novos que não se mantêm, são apagados
334 LibertarVector(geracaoNova);
335 // tudo pronto para uma nova geração
336 }
337 // não há mais gerações, apagar tudo
338 LibertarVector(geracaoAntiga);
339 return melhor->custo;
340}

386{
387 TVector<TPonto> populacao, pais, descendentes;

512 int& minDist, int& maxDist, int& avgDist, int& melhorPior)
513{
514 // processar todos os pares
515 // Futuro: caso POPULACAO seja grande, processar apenas alguns pares
516 int contagem = 0, melhor = 0, pior = populacao.Count() - 1;
517 minDist = INT_MAX;
518 maxDist = 0;
519 avgDist = 0;
520 for (int i = 0; i < populacao.Count(); i++) {
521 if (populacao[i]->custo < populacao[melhor]->custo)
522 melhor = i;
523 if (populacao[i]->custo > populacao[pior]->custo)
524 pior = i;
525 for (int j = i + 1; j < populacao.Count(); j++) {
526 int dist = populacao[i]->Distancia(populacao[j]);
527 if (dist < minDist)
528 minDist = dist;
529 if (dist > maxDist)
530 maxDist = dist;
531 avgDist += dist;
532 contagem++;
533 }
534 }
535 melhorPior = populacao[melhor]->Distancia(populacao[pior]);
536 if (contagem > 0)
537 avgDist /= contagem;
538 else
539 minDist = avgDist = maxDist = 0;

556 // selecionar os pais, de acordo com o método escolhido
557 int descendentes = (Parametro(PERC_DESCENDENTES) * populacao.Count()) / 100;
558 int pop = populacao.Count();
559 if (descendentes < 1)
560 descendentes = 1;
561 else if (descendentes > pop)
562 descendentes = pop;
563
564 Debug(COMPLETO, false, "\n │ ├─┬─── FASE 🎯 Selecionar %d 🧑‍🤝‍🧑 pais ───── ", descendentes);
565 pais = {};
566 if (Parametro(SELECAO) == 1) { // roleta
567 // roleta implementada como Stochastic Universal Sampling (SUS)
568 TVector<int> id, peso;
569 TVector<float> probabilidades;
570 int pressao = Parametro(PRESSAO);
571 Debug(COMPLETO, false, "\n │ │ ├───── Roleta, pressão %d ───── ", pressao);
572 OrdemValor(populacao, id);
573 peso.Count(id.Count());
574 peso.Reset(0);
575 for (int i = 0; i < id.Count(); i++)
576 peso[id[i]] = id.Count() - i;
577 for (int i = 0; i < populacao.Count(); i++) {
578 // calcular a probabilidade de um elemento ser selecionado
579 // utilizando o método Ranking Selection
580 probabilidades += ((2.0f - (float)pressao / 100.0f) / pop +
581 (2.0f * (peso[i] - 1) * ((float)pressao / 100.0f - 1.0f)) / (float)(pop * (pop - 1)));
582 if (i > 0)
583 probabilidades[i] += probabilidades[i - 1];
584 }
585 probabilidades.Last() = 1;
586 if (Parametro(NIVEL_DEBUG) >= COMPLETO) {
587 TVector<int> prob;
588 for (int i = 0; i < probabilidades.Count(); i++)
589 prob += (int)((i == 0 ? probabilidades[i] :
590 probabilidades[i] - probabilidades[i - 1]) * 1000 + 0.5);
591 DebugTabela(COMPLETO, prob, "100%", " │ │ │ ", -1);
592 }
593 // escolher os pais (roleta implementado como SUS)
594 float valor = (float)(TRand::rand() % 10000) / (descendentes * 10000.0f);
595 for (int escolhido = 0; pais.Count() < descendentes; escolhido++) {
596 while (pais.Count() < descendentes && valor <= probabilidades[escolhido]) {
597 pais += populacao[escolhido];
598 valor += 1.0f / (float)descendentes;
599 }
600 }
601 }
602 else if (Parametro(SELECAO) == 3) { // uniforme
603 Debug(COMPLETO, false, "\n │ │ ├───── Uniforme ───── ");
604 for (int i = 0; i < descendentes; i++)
605 pais += populacao.Random();
606 }
607 else if (Parametro(SELECAO) == 2) { // torneio
608 int tamanho = Parametro(TAMANHO_TORNEIO);
609 int probMelhor = Parametro(PROB_MELHOR_TORNEIO);
610 Debug(COMPLETO, false, "\n │ │ ├───── Torneio, tamanho %d, probabilidade melhor %d ───── ",
611 tamanho, probMelhor);
612 for (int i = 0; i < descendentes; i++) {
613 TVector<TPonto> competidores;
614 TVector<int> id;
615 for (int j = 0; j < tamanho; j++)
616 competidores += populacao.Random();
617 // ordenar por custo
618 OrdemValor(competidores, id);
619 // escolher melhor ou segundo melhor
620 if ((TRand::rand() % 100) < (unsigned)probMelhor)
621 pais += competidores[id[0]];
622 else
623 pais += competidores[id[1 % tamanho]];
624 }
625 }
626 // reportar quantas vezes cada elemento foi pai
627 if (Parametro(NIVEL_DEBUG) >= COMPLETO) {
628 TVector<int> pai;
629 Debug(COMPLETO, false, "\n │ │ ├───── Número de seleções ───── ");
630 pai.Count(populacao.Count());
631 pai.Reset(0);
632 for (auto elemento : pais) {
633 int id = populacao.Find(elemento);
634 if (id >= 0)
635 pai[id]++;
636 }
637 DebugTabela(COMPLETO, pai, "#Pai", " │ │ │ ", -1);
638 }
639 Debug(COMPLETO, false, "\n │ │ └──────────────────────────────────── ");
640 return pais;
641}

643TVector<TPonto> TProcuraMelhorativa::ReproduzirAE(TVector<TPonto>& pais, TVector<TPonto>& popoulacao) {
644 TVector<TPonto> descendentes;
645 int cruzamentos = 0, mutacoes = 0;
646 int melhor, pior, igual; // estatística do avanço da nova geração
647 TVector<int> custoPais, custoFilhos;
648 Debug(COMPLETO, false, "\n │ ├─┬─── FASE Reproduzir %d pais ───── ", pais.Count());
649 Debug(COMPLETO, false, "\n │ │ ├───── Pais (🧑‍🤝‍🧑) ───── ");
650 pais.RandomOrder();
651 if (Parametro(NIVEL_DEBUG) >= COMPLETO) {
652 TVector<int> paiID;
653 for (auto elemento : pais) {
654 int id = popoulacao.Find(elemento);
655 if (id >= 0)
656 paiID += (id + 1);
657 }
658 DebugTabela(COMPLETO, paiID, " 🧑‍🔬 ", " │ │ │ ", popoulacao.Count(), true);
659 }
660 while (!pais.Empty()) {
661 TPonto pai = pais.Pop();
662 custoPais += pai->custo;
663
664 if (pais.Empty()) {
665 // não há mãe, apenas copiar o pai
666 descendentes += pai->Duplicar();
667 }
668 else {
669 TPonto mae = pais.Pop();
670 custoPais += mae->custo;
671 if (TRand::rand() % 100 < (unsigned)Parametro(PROB_CRUZAR)) {
672 // gerar um novo individuo por cruzamento
673 TPonto filho = pai->Duplicar();
674 filho->Cruzamento(pai, mae);
675 descendentes += filho;
676 filho = mae->Duplicar();
677 filho->Cruzamento(mae, pai);
678 descendentes += filho;
679 cruzamentos++;
680 }
681 else {
682 // copia o elemento sem alterações
683 descendentes += pai->Duplicar();
684 descendentes += mae->Duplicar();
685 }
686 }
687 }
688 custoPais.Invert();
689 // mutar e avaliar descendentes
690 for (auto descendente : descendentes) {
691 // mudar o novo elemento, dependente da probabilidade
692 if (TRand::rand() % 100 < (unsigned)Parametro(PROB_MUTAR)) {
693 descendente->Mutar();
694 mutacoes++;
695 }
696 // avaliar o valor do elemento
697 descendente->Avaliar();
698 custoFilhos += descendente->custo;
699 VerificaMelhor(descendente);
700 }
701 int maiorCusto = 0;
702 for (auto custo : custoPais)
703 if (maiorCusto < custo)
704 maiorCusto = custo;
705 for (auto custo : custoFilhos)
706 if (maiorCusto < custo)
707 maiorCusto = custo;
708
709 melhor = pior = igual = 0;
710 for (int i = 0; i < custoPais.Count(); i += 2) {
711 int paiMelhor, paiPior;
712 paiMelhor = paiPior = custoPais[i];
713 if (paiMelhor > custoPais[i + 1])
714 paiMelhor = custoPais[i + 1];
715 if (paiPior < custoPais[i + 1])
716 paiPior = custoPais[i + 1];
717 for (int j = i; j <= i + 1; j++)
718 if (custoFilhos[j] > paiPior)
719 pior++;
720 else if (custoFilhos[j] < paiMelhor)
721 melhor++;
722 else
723 igual++;
724 }
725
726 Debug(COMPLETO, false, "\n │ │ ├───── Pais (💰) ───── ");
727 DebugTabela(COMPLETO, custoPais, " 🧑‍🔬 ", " │ │ │ ", 1000000 + maiorCusto, true);
728 Debug(COMPLETO, false, "\n │ │ ├───── Filhos (💰) 🧬%d ✨%d ───── 📈%d ➖%d 📉%d",
729 cruzamentos, mutacoes, melhor, igual, pior);
730 DebugTabela(COMPLETO, custoFilhos, " 🧑‍🔬 ", " │ │ │ ", 1000000 + maiorCusto, true);
731 Debug(COMPLETO, false, "\n │ │ └──────────────────────────────────── ");
732 return descendentes;
733}

735TVector<TPonto> TProcuraMelhorativa::SelecionarSobreviventesAE(TVector<TPonto>& populacao, TVector<TPonto>& descendentes) {
736 TVector<TPonto> elite;
737 TVector<int> idElite;
738 int nElite = Parametro(ELITISMO);
739 int imigrantes = Parametro(IMIGRANTES);
740 int melhorDescendente = INT_MAX;
741 Debug(COMPLETO, false, "\n │ ├─┬─── FASE 🛡️ Sobrevivência ───── ");
742 if (nElite > 0) {
743 // 1. Copiar os N melhores da população atual
744 TVector<int> id;
745 OrdemValor(populacao, id); // melhor primeiro
746 for (int i = 0; i < nElite && i < populacao.Count(); i++) {
747 elite += populacao[id[i]]->Duplicar();
748 elite.Last()->custo = populacao[id[i]]->custo;
749 idElite += id[i];
750 }
751 // encontrar o melhor descendente
752 melhorDescendente = MelhorCusto(descendentes);
753 }
754
755 if (Parametro(SOBREVIVENCIA) == 1) { // idade
756 Debug(COMPLETO, false, "\n │ │ ├───── ⏳ Idade ───── ");
757 // remover os mais velhos (os que estão mais tempo na população)
758 for (int i = 0; i < descendentes.Count() && i < populacao.Count(); i++) {
759 delete populacao[i];
760 populacao[i] = NULL;
761 }
762 populacao -= NULL;
763 populacao += descendentes;
764 descendentes = {};
765 }
766 else if (Parametro(SOBREVIVENCIA) == 2) { // substituir piores
767 TVector<int> custos;
768 Debug(COMPLETO, false, "\n │ │ ├───── Substituir piores ───── ");
769 populacao += descendentes;
770 descendentes = {};
771 if (Parametro(NIVEL_DEBUG) >= COMPLETO) {
772 for (auto individuo : populacao)
773 custos += individuo->custo;
774 Debug(COMPLETO, false, "\n │ │ ├───── Custos em ambas gerações ───── ");
775 DebugTabela(COMPLETO, custos, "Ind", " │ │ │ ");
776 }
777 TVector<int> id;
778 OrdemValor(populacao, id); // ordena por custo (melhor primeiro)
779 TVector<TPonto> novaPopulacao;
780 for (int i = 0; i < Parametro(POPULACAO); i++) {
781 novaPopulacao += populacao[id[i]];
782 populacao[id[i]] = NULL;
783 }
784 LibertarVector(populacao);
785 populacao = novaPopulacao;
786 }
787 else if (Parametro(SOBREVIVENCIA) == 3) { // round-robin
788 // cada elemento compete com q outros, os que perdem mais são eliminados
789 int q = Parametro(Q_ROUND_ROBIN); // número de competidores
790 TVector<int> perdas, custos;
791 Debug(COMPLETO, false, "\n │ │ ├───── Round Robin (q: %d) ───── ", q);
792 populacao += descendentes;
793 descendentes = {};
794 if (Parametro(NIVEL_DEBUG) >= COMPLETO) {
795 for (auto individuo : populacao)
796 custos += individuo->custo;
797 Debug(COMPLETO, false, "\n │ │ ├───── Custos em ambas gerações ───── ", q);
798 DebugTabela(COMPLETO, custos, "Ind", " │ │ │ ");
799 }
800 perdas.Count(populacao.Count());
801 perdas.Reset(0);
802 for (int i = 0; i < populacao.Count(); i++) {
803 TPonto atual = populacao[i];
804 for (int j = 0; j < q; j++) {
805 TPonto oponente;
806 do {
807 oponente = populacao.Random();
808 } while (oponente == atual && populacao.Count() > 1);
809 if (atual->custo > oponente->custo)
810 perdas[i]++;
811 }
812 }
813 Debug(COMPLETO, false, "\n │ │ ├───── Número de perdas ───── ");
814 DebugTabela(COMPLETO, perdas, "Perd", " │ │ │ ");
815 // ordenar por número de perdas
816 TVector<int> id;
817 perdas.Sort(&id);
818 TVector<TPonto> novaPopulacao;
819 for (int i = 0; i < Parametro(POPULACAO); i++) {
820 novaPopulacao += populacao[id[i]];
821 populacao[id[i]] = NULL;
822 }
823 LibertarVector(populacao);
824 populacao = novaPopulacao;
825 }
826
827 if (imigrantes > 0) {
828 // 3. Adicionar novos elementos aleatórios
829 Debug(COMPLETO, false, "\n │ │ ├───── 🚶‍♂️🌍 Imigrantes ");
830 for (int i = 0; i < imigrantes; i++) {
831 // remover um aleatório para dar lugar a um imigrante
832 int idx = TRand::rand() % populacao.Count();
833 delete populacao[idx];
834 populacao[idx] = NULL;
835 Debug(COMPLETO, false, " %d✖ →🆕", idx + 1);
836 }
837 Debug(COMPLETO, false, " ───── ");
838 }
839 populacao -= NULL;
840
841 if (nElite > 0) {
842 // 2. Garantir que os elites estão presentes
843 // (apenas se forem melhores que o melhor descendente)
844 bool primeiro = false;
845 for (int i = 0; i < elite.Count(); i++) {
846 if (elite[i]->custo < melhorDescendente) {
847 // remover um aleatório para dar lugar ao elite
848 int idx = TRand::rand() % populacao.Count();
849 delete populacao[idx];
850 populacao[idx] = elite[i];
851 if (!primeiro) {
852 Debug(COMPLETO, false, "\n │ │ ├───── 👑 Elite %d→%d",
853 idElite[i] + 1, idx + 1);
854 primeiro = true;
855 }
856 else
857 Debug(COMPLETO, false, " %d→%d",
858 idElite[i] + 1, idx + 1);
859 }
860 else
861 delete elite[i]; // já não é necessário
862 }
863 if (primeiro)
864 Debug(COMPLETO, false, " ───── ");
865 }
866 Debug(COMPLETO, false, "\n │ │ └──────────────────────────────────── ");
867 return populacao;
868}

870TVector<TPonto> TProcuraMelhorativa::AplicarDiversidadeAE(TVector<TPonto>& populacao)
871{
872 int distMinima = Parametro(DIST_MINIMA);
873 if (Parametro(DIVERSIDADE) == 2) { // avaliação partilhada
874 TVector<int> id, penalizados;
875 int count = 0;
876 for (int i = 0; i < populacao.Count(); i++)
877 id += i;
878 Debug(COMPLETO, false, "\n │ ├───── FASE Diversidade - avaliação partilhada ───── ");
879 for (int j = 0; j < populacao.Count(); j++) {
880 if (id.Count() > 20)
881 id.RandomOrder();
882 for (int i = 0; i < id.Count() && i < 20; i++)
883 if (j != id[i] &&
884 populacao[j]->Distancia(populacao[id[i]]) <= distMinima) {
885 populacao[j]->custo++; // soma 1 por cada elemento muito parecido
886 populacao[id[i]]->custo++;
887 count++;
888 penalizados += (id[i] + 1);
889 penalizados += (j + 1);
890 }
891 }
892 if (!penalizados.Empty()) {
893 penalizados.BeASet();
894 Debug(COMPLETO, false, " (%d penalizações aplicadas)", count);
895 DebugTabela(COMPLETO, penalizados, "Ind", " │ │ ");
896 }
897 Debug(COMPLETO, false, "\n │ └────────────────────────────────────── ");
898 }
899 else if (Parametro(DIVERSIDADE) == 3) { // limpeza
900 TVector<int> id, remover;
901 for (int i = 0; i < populacao.Count(); i++)
902 id += i;
903 Debug(COMPLETO, false, "\n │ └───── FASE Diversidade - limpeza ───── ");
904 for (int j = 0; j < populacao.Count(); j++) {
905 if (id.Count() > 20)
906 id.RandomOrder();
907 for (int i = 0; i < id.Count() && i < 20; i++)
908 if (j != id[i] &&
909 populacao[j]->Distancia(populacao[id[i]]) <= distMinima)
910 {
911 remover += id[i];
912 //remover += j; // apenas um
913 id -= id[i];
914 id -= j;
915 break;
916 }
917 }
918 for (auto i : remover) {
919 delete populacao[i];
920 populacao[i] = NULL;
921 Debug(COMPLETO, false, " 🧹✖ %d", i + 1);
922 }
923 populacao -= NULL;
924 }
925 return populacao;
926}
