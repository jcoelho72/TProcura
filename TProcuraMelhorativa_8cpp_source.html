78int TProcuraMelhorativa::EscaladaDoMonte()
79{
80 bool movePrimeiro = (parametro[movePrimeiroEM].valor == 1);
81 TPonto atual = this; // estado atual
82 TPonto melhor = this; // melhor estado para todas as escaladas
83 int procuras = 0;
84 TPonto solucao;
85 NovaSolucao();
86 melhor->custo = atual->custo = Avaliar(); // avaliar o proprio estado
87 // estado atual distinto do melhor
88 atual = solucao = (TProcuraMelhorativa*)Duplicar();
89 while(!melhor->Parar()) {
90 // iniciar uma escalada
91 // ponto de partida uma solução aleatória (local no espaço de soluções completas)
92 solucao->NovaSolucao();
93 atual->custo = solucao->Avaliar();
94 DebugInicioEM(++procuras, solucao);
95
96 while(!melhor->Parar()) {
97 // ver de entre os vizinhos, qual o melhor, para avançar um degrau
98 TVector<TPonto> vizinhos;
99 bool alterado = false;
100 solucao->Vizinhanca(vizinhos);
101 if(movePrimeiro) {
102 // move logo que encontre um vizinho melhor
103 // (pode estar a ser ignorados vizinhos melhores)
104 while (vizinhos.Count() > 0) {
105 ((TProcuraMelhorativa*)vizinhos.Last())->Avaliar();
106 if (atual->custo > vizinhos.Last()->custo) {
107 solucao = MelhorAtual(atual, vizinhos, vizinhos.Count() - 1);
108 VerificaMelhor(melhor, atual);
109 // novos vizinhos a partir do estado atual são apagados
110 LibertarVector(vizinhos);
111 alterado = true;
112 }
113 else {
114 delete vizinhos.Last();
115 vizinhos.Pop();
116 }
117 }
118 } else {
119 // analisa todos os vizinhos e avança apenas para o melhor vizinho
120 int melhorValor, melhorIndice;
121 CalcularAvaliacoes(vizinhos, melhorValor, melhorIndice);
122 // trocar caso melhore
123 if(atual->custo > melhorValor) {
124 solucao = MelhorAtual(atual, vizinhos, melhorIndice);
125 VerificaMelhor(melhor, atual);
126 // novos vizinhos a partir do estado atual
127 alterado = true;
128 }
129 // tudo explorado, libertar
130 LibertarVector(vizinhos);
131 }
132 // final da análise da vizinhança, ver se se subiu alguma coisa
133 if (!alterado) {
134 // significa que estamos num óptimo local, no topo do moonte
135 DebugOptimoLocal(solucao);
136 break; // recomeçar nova escalada
137 }
138 else
139 DebugPassoEM(solucao);
140 }
141 }
142 //delete atual;
143 delete solucao;
144
145 return melhor->custo;
146}

220// retorna a avaliacao do resultado final
221// parametros para a mutacao e cruzamento podem ser dados em variaveis globais
222int TProcuraMelhorativa::AlgoritmoGenetico()
223{
224 int populacao = parametro[populacaoAG].valor;
225 int probablidadeMutacao = parametro[probMutacaoAG].valor;
226 int distanciaMinima = parametro[distMinimaAG].valor;
227 TVector<TPonto> geracaoAntiga, geracaoNova;
228 TVector<int> id;
229 TPonto melhor = this; // melhor estado para toda a procura
230 NovaSolucao();
231 melhor->Avaliar(); // avaliar o proprio
232 if(parametro[nivelDebug].valor > 1)
233 printf("\nPopulação inicial (%d elementos)",populacao);
234 // construir a geracao inicial
235 for (int i = 0; i < populacao; i++) {
236 geracaoAntiga.Add((TPonto)Duplicar());
237 geracaoAntiga.Last()->NovaSolucao();
238 geracaoAntiga.Last()->Avaliar();
239 VerificaMelhor(melhor, geracaoAntiga.Last());
240 }
241 while(!melhor->Parar()) {
242 // iniciar uma nova geração, com base na antiga
243 expansoes++;
244
246 // cruzar conforme o custo
248 int minimo,maximo,total;
249 TVector<int> probabilidade;
250 total=0;
251 ObterExtremos(geracaoAntiga, minimo, maximo);
252 DebugPassoAG(geracaoAntiga.Count(), minimo, maximo);
253 // calcular a probabilidade de um elemento ser selecionado
254 for (int i = 0; i < geracaoAntiga.Count(); i++) {
255 // cada elemento tem de ter sempre probabilidade de ser escolhido
256 // por isso adiciona-se 1 para que o custo maximo tenha probabilidade minima
257 probabilidade.Add(maximo - geracaoAntiga[i]->custo + 1);
258 total += probabilidade.Last();
259 }
260 // gerar novos elementos. gera-se o doubro para poder seleccionar os melhores
261 while (geracaoNova.Count() < 2 * populacao) {
262 // escolher dois individuos aleatoriamente de acordo com as probabilidades
263 int pai, mae, mutou=0;
264 Selecao(pai, mae, probabilidade, total);
265 // gerar um novo individuo por cruzamento
266 geracaoNova.Add((TPonto)Duplicar());
267 geracaoNova.Last()->Cruzamento(geracaoAntiga[pai], geracaoAntiga[mae]);
268 // mudar o novo elemento, dependente da probabilidade
269 if (TRand::rand() % 100 < (unsigned)probablidadeMutacao) {
270 mutou = 1;
271 geracaoNova.Last()->Mutar();
272 }
273 // avaliar o valor do elemento
274 geracaoNova.Last()->Avaliar();
275 DebugCruzamentoAG(
276 geracaoAntiga[pai]->custo,
277 geracaoAntiga[mae]->custo,
278 geracaoNova.Last()->custo,
279 mutou);
280 VerificaMelhor(melhor, geracaoNova.Last());
281 }
282
283 // a nova geração já tem elementos, apagar a antiga
284 LibertarVector(geracaoAntiga);
285 // seleccao: os melhores, com base no valor
286 OrdemValor(geracaoNova, id);
287 // remover os elementos que são muito parecidos
288 for (int i = 0; i < populacao && i < geracaoNova.Count(); i++) {
289 // verificar se ha outro elemento muito parecido la
290 if (i > 0 && distanciaMinima > 0) {
291 bool diferente = true;
292 for (int j = 0; j < geracaoAntiga.Count(); j++)
293 if (geracaoAntiga[j]->Distancia(geracaoNova[id[i]]) < distanciaMinima) {
294 diferente = false;
295 break;
296 }
297 if(!diferente)
298 continue;
299 }
300 // este elemento fica, passando para a geração antiga
301 geracaoAntiga.Add(geracaoNova[id[i]]);
302 geracaoNova[id[i]] = NULL;
303 }
304 // todos os novos que não se mantêm, são apagados
305 LibertarVector(geracaoNova);
306 // tudo pronto para uma nova geração
307 }
308 // não há mais gerações, apagar tudo
309 LibertarVector(geracaoAntiga);
310 return melhor->custo;
311}
