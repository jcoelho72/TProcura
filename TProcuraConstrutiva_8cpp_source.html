102void TProcuraConstrutiva::Sucessores(TVector<TNo>& sucessores) {
103 if (memoriaEsgotada)
104 return;

106 for (int i = 0; i < sucessores.Count(); i++) {
107 sucessores[i]->pai = this;

110 }
111 // verificação de estados repetidos

118 }
119 }
120 else if (Parametro(ESTADOS_REPETIDOS) == ASCENDENTES) {
121 // verificar se o estado já foi gerado na árvore de procura
122 // remover todos os estados que ja tenham sido expandidos neste ramo
123 for (int i = 0; i < sucessores.Count(); i++) {
124 TNo avo = this;
125 while (avo != NULL && avo->Distinto(sucessores[i]))
126 avo = avo->pai;
127 // estado é igual a um avô, não interessa
128 if (avo != NULL) {
129 delete sucessores[i];
130 sucessores[i] = NULL;
131 }
132 }
133 }
134 sucessores -= NULL;
135 if (Parametro(BARALHAR_SUCESSORES) == 1)
136 sucessores.RandomOrder();
137 expansoes++;

152// Procura em Largura Primeiro: expande primeiro o estado gerado não expandido mais antigo
153// retorna o valor da solução e coloca o caminho no vector (se calcularCaminho=true), ou -1 caso não encontre solucao
154// limite é o número de estados gerados não expandidos, que não pode ultrapassar esse limite
155// os que ultrapassarem são deitados fora (se 0 este limite não importa, podendo haver problemas de memória)
156int TProcuraConstrutiva::LarguraPrimeiro(int limite)
157{
158 TVector<TNo> lista; // lista de nós gerados na árvore de procura (expandidos ou não)
159 lista += this; // nó inical: estado, custo
160 for (int i = 0; i < lista.Count() && !Parar(); i++) {
161 // processar o próximo estado não expandido (estão expandidos todos os anteriores a i)
162 lista[i]->DebugPasso();
163 // expansão deste estado
164 TVector<TNo> sucessores;
165 lista[i]->Sucessores(sucessores);
166 // garantir que os limites são respeitados, para evitar problemas de memória
167 VerificaLimites(limite, lista.Count() - i, sucessores);
168 // inserir tudo no final da lista
169 for (int j = 0; j < sucessores.Count(); j++) {
170 lista += sucessores[j];
171 // verificar se é um estado objetivo, a solução é completa nesse caso
172 // teste na geração e não na expansão
173 if (lista.Last()->SolucaoCompleta()) {
174 LibertarVector(sucessores, -1, j);
175 return ObjetivoAlcancado(lista.Count() - 1, lista); // Sucesso! Terminar a procura e retornar
176 }
177 }
178
179 if (Parametro(NIVEL_DEBUG) > DETALHE)
180 lista[i]->DebugSucessores(sucessores);
181
182 // Nao se pode libertar estados ja expandidos porque nao se sabe se
183 // os pais sao necessarios ou nao.
184 // todos os estados antes de i, são os estados gerados e expandidos (fechados)
185 // todos os estados após i são estados gerados mas não expandidos (abertos)
186 }
187 // nada feito, libertar tudo excepto o estado inicial que é este objeto, e retornar
188 LibertarVector(lista, 0);
189 return -1; // falhou
190}

208void TProcuraConstrutiva::CalculaCaminho(bool completa) {
209 TNo atual = this;
210 // limpar o caminho anterior (caso do BnB que obtém mais que uma solução)
211 while (!caminho.Empty())
212 delete caminho.Pop();
213 caminho += atual->Duplicar();
214 caminho.Last()->custo = atual->custo;
215
216 // apenas se a lista estiver completa, há a garantia dos pais não terem sido apagados
217 if (completa) {
218 // obter caminho para a solucao
219 while (atual->pai != NULL) {
220 atual = atual->pai;
221 caminho += atual->Duplicar();
222 caminho.Last()->custo = atual->custo;
223 }
224 }
225 caminho.Invert();

270// Procura em Profundidade Primeiro: expande primeiro o estado gerado não expandido mais novo
271// caso o nivel=-1, e feita uma procura em profunidade normal
272// caso o nivel>0, e feita uma procura em profundidade limitada
273// caso o nivel=0, e feita uma procura em profundidade iterativa, sem limite
274// versão recursiva
275int TProcuraConstrutiva::ProfundidadePrimeiro(int nivel)
276{
277 DebugChamada();
278 if (nivel == 0) { // metodo iterativo
279 int resultado = -1;
280 do {
281 // limpar hashtable: estados gerados no nível anterior não devem impedir nova geração
282 LimparHT();
283 DebugIteracao(nivel + 1);
284 // chamar a profundidade nível 1, e se não resolver, o nível 2, e assim sucessivamente
285 resultado = ProfundidadePrimeiro(++nivel);
286 } while (resultado == -1 && !Parar());
287 return resultado;
288 }
289
290 // metodo normal
291 // verificar se o estado atual é objetivo, ou seja, a solução parcial é já completa
292 if (SolucaoCompleta())
293 return SolucaoEncontrada();
294
295 if ((nivel > 1 || nivel < 0) && !Parar()) {
296 // caso o nível seja superior a 1 ou sem limite, expandir o estado atual
297 TVector<TNo> sucessores;
298 Sucessores(sucessores);
299 // tentar todos os sucessores, um de cada vez
300 for (int i = 0; i < sucessores.Count(); i++) {
301 DebugExpansao(i, sucessores.Count());
302 // chamada recursiva, reduzindo o nível
303 if (sucessores[i]->ProfundidadePrimeiro(nivel - 1) >= 0)
304 // este sucessor resolveu o problema, retornar
305 return SolucaoParcial(i, sucessores);
306 }
307 // nenhum dos sucessores resolveu o problema
308 DebugCorte(sucessores.Count());
309 LibertarVector(sucessores);
310 }
311 else
312 DebugCorte();
313
314 // falha na procura neste nó, ou não há estado objetivo a partir daqui
315 // ou atingiu-se o limite, pelo que temos de retornar -1
316 return -1;
317}

371// Algoritmo Melhor Primeiro: expande primeiro o estado gerado não expandido com melhor heurística
372// versão recursiva, idêntico a ProcuraPrimeiro()
373int TProcuraConstrutiva::MelhorPrimeiro(int nivel)
374{
375 DebugChamada();
376 if (SolucaoCompleta())
377 return SolucaoEncontrada();
378
379 if ((nivel <= 0 || nivel > 1) && !Parar()) {
380 TVector<TNo> sucessores;
381 TVector<int> id; // índice para ordenar os sucessores por heurística
382 Sucessores(sucessores);
383 CalcularHeuristicas(sucessores, &id); // id fica ordenado por heurística
384 // processar todos os sucessores da mesma forma que a procura em profundidade,
385 // mas utilizando a ordem id, por heurística
386 for (int i = 0; i < id.Count(); i++) {
387 DebugExpansao(i, id.Count());
388 if (sucessores[id[i]]->MelhorPrimeiro(nivel - 1) >= 0)
389 return SolucaoParcial(id[i], sucessores);
390 }
391 DebugCorte(sucessores.Count());
392 LibertarVector(sucessores);
393 }
394 else
395 DebugCorte();
396 return -1;

400// Algoritmo IDAStar: procura em profundidade limitada a um upperBound
401// idêntico a MelhorPrimeiro() mas cortando quando upperBound já não consegue ser obtido
402int TProcuraConstrutiva::IDAStar(int upperBound)
403{
404 DebugChamada();
405 if (upperBound == 0) { // parte iterativa
406 int resultado = -1;
407 // primeiro valor para o lower bound, a heurística no nó raiz
408 lowerBound = Heuristica(); // o custo é zero atualmente
409 if (lowerBound == 0)
410 lowerBound++;
411 do {
412 // limpar hashtable: estados gerados no nível anterior não devem impedir nova geração
413 LimparHT();
414 DebugIteracao(lowerBound);
415 // ver se há uma solução com este valor
416 resultado = IDAStar(lowerBound);
417 // o valor de lowerBound é atualizado, para utilizar na próxima iteração se necessário
418 } while (resultado == -1 && !Parar());
419 return resultado;
420 }
421
422 if (SolucaoCompleta())
423 return SolucaoEncontrada();
424
425 if (!Parar()) {
426 TVector<TNo> sucessores;
427 TVector<int> id; // índice para ordenar os sucessores por heurística
428 Sucessores(sucessores);
429 CalcularHeuristicas(sucessores, &id, true); // id fica ordenado por LB
430 // processar todos os sucessores da mesma forma que a procura em profundidade,
431 // mas utilizando a ordem id, por heurística
432 for (int i = 0; i < id.Count(); i++) {
433 int atual = sucessores[id[i]]->LowerBound();
434 DebugExpansao(i, id.Count());
435 if (atual > upperBound) {
436 // acima do permitido nesta iteração
437 if (lowerBound == upperBound || lowerBound > atual)
438 lowerBound = atual;
439 DebugCorte(); // estado cortado, não expandido
440 }
441 else {
442 if (sucessores[id[i]]->IDAStar(upperBound) >= 0)
443 return SolucaoParcial(id[i], sucessores);
444 }
445 }
446 DebugCorte(sucessores.Count());
447 LibertarVector(sucessores);
448 }
449 return -1;
450}

453// Algoritmo BranchAndBound:
454// idêntico a MelhorPrimeiro(), mas continua após a primeira solução
455// explora apenas os estados em que f(n) < atual solução (LB<UB)
456int TProcuraConstrutiva::BranchAndBound(int upperBound)
457{
458 DebugChamada();
459 if (SolucaoCompleta())
460 return SolucaoEncontrada(true);
461
462 if (!Parar()) {
463 TVector<TNo> sucessores;
464 TVector<int> id; // índice para ordenar os sucessores por heurística
465 Sucessores(sucessores);
466 CalcularHeuristicas(sucessores, &id, true); // id fica ordenado por LB
467 // processar todos os sucessores, mas apenas se LB<UB
468 for (int i = 0; i < id.Count(); i++) {
469 DebugExpansao(i, id.Count());
470 if (upperBound && sucessores[id[i]]->LowerBound() >= upperBound) {
471 DebugCorte(); // estado cortado, e os seguintes não serão expandidos
472 break;
473 }
474 int resultado = sucessores[id[i]]->BranchAndBound(upperBound);
475 // mesmo que exista um resultado, continuar à procura de melhor
476 if (resultado > 0 && (!upperBound || resultado < upperBound))
477 upperBound = resultado;
478 }
479 DebugCorte(sucessores.Count());
480 LibertarVector(sucessores);
481 }
482 return (solucao != NULL ? solucao->custo : -1);
483}

485void TProcuraConstrutiva::CalcularHeuristicas(
486 TVector<TNo>& sucessores, TVector<int>* id, bool sortLB)
487{
488 TVector<int> heuristicas;

502// Idêntico a CustoUniforme()
503int TProcuraConstrutiva::AStar(int limite)
504{

765void TProcuraConstrutiva::Explorar() {
766 TVector<TNo> sucessores;
767 int opcao = 0;
768 LimparEstatisticas();
769 do {
770 caminho += Duplicar();
771 if (caminho.Count() > 1)
772 caminho.Last()->custo += caminho[caminho.Count() - 2]->custo;
773 else
774 caminho.Last()->custo = 0;
775 heuristica = Heuristica();
776 Sucessores(sucessores);
777 CalcularHeuristicas(sucessores);
778 // linha com informação
779 DebugEstado();
780 Debug();
781 DebugSucessores(sucessores);
782 if (sucessores.Empty()) {
783 TProcura::Mensagem("ℹ️ Informação", "Sem sucessores.");
784 opcao = 0;
785 }
786 else {
787 char str[BUFFER_SIZE];
788 printf("\n🔍 Sucessor [1-%d, ação(ões), exe]: ", sucessores.Count());
789 if (!fgets(str, BUFFER_SIZE, stdin))
790 str[0] = 0;
791 opcao = atoi(str);
792 if (opcao == 0 && strlen(str) > 1) {
793 char* token;
794 opcao = sucessores.Count() + 1;
795 token = strtok(str, " \t\n\r");
796 // executar algoritmo
797 if (strcmp(token, "exe") == 0) {
798 TVector<TNo> backup;
799 backup = caminho;
800 caminho = {};
801 LimparEstatisticas();
802 int resultado = 0;
803 switch (resultado = ExecutaAlgoritmo()) {
804 case -1: TProcura::Mensagem("ℹ️ Informação", "Impossível"); break;
805 case -2: TProcura::Mensagem("", "Não resolvido"); break;
806 default: TProcura::Mensagem("✅ Sucesso", "Resolvido (%d)", resultado); break;
807 }
808 tempo = Cronometro(CONT_ALGORITMO);
809 if (solucao != NULL) {
810 Copiar(solucao);
811 delete backup.Pop();
812 backup += caminho;
813 caminho = backup;
814 caminho.Pop(); // este último estado será adicionado
815 }
816 else
817 caminho = backup;
818 }
819 else {
820 int nAcoes = 0;
821 do {
822 // executar a ação
823 if (Acao(token))
824 nAcoes++;
825 else {
826 TProcura::Mensagem("", "Ação %s inválida.", token);
827 break;
828 }
829 token = strtok(NULL, " \t\n\r");
830
831 if (token != NULL) {
832 // há outra ação, é preciso atualizar o caminho
833 caminho += Duplicar();
834 if (caminho.Count() > 1)
835 caminho.Last()->custo += caminho[caminho.Count() - 2]->custo;
836 else
837 caminho.Last()->custo = 0;
838 }
839
840 } while (token != NULL);
841 if (nAcoes > 0)
842 TProcura::Mensagem("✅ Sucesso", "Executadas %d ações.", nAcoes);
843 }
844 }
845 }
846 if (opcao > 0 && opcao <= sucessores.Count()) {
847 Copiar(sucessores[opcao - 1]);
848 TProcura::Mensagem("✅ Sucesso", "Ação executada.");
849 }
850 LibertarVector(sucessores);
851 } while (opcao != 0);
852}

854unsigned int TProcuraConstrutiva::Hash() {
855 // utilizando FNV-1 hash (http://www.isthe.com/chongo/tech/comp/fnv/)
856 uint32_t h = 2166136261;
857 Codifica(estadoCodHT);
858 for (int i = 0; i < OBJETO_HASHTABLE && estadoCodHT[i]; i++) {
859 uint64_t valor = estadoCodHT[i];
860 // processar cada octeto
861 // quando o valor fica 0, já não interessa
862 // já que existindo dados, são distintos de 0
863 for (int i = 0; i < 8 && valor; i++) {
864 unsigned char octeto = valor & 255;
865 valor >>= 8;
866 h *= 16777619;
867 h ^= octeto;
868 }
869 }
870 return h % TAMANHO_HASHTABLE;
871}

873void TProcuraConstrutiva::LimparHT() {
874 if (Parametro(ESTADOS_REPETIDOS) == GERADOS) {
875 if (Parametro(NIVEL_DEBUG) >= ATIVIDADE) {
876 int usado = 0; // contar para calcular taxa de ocupação
877 for (int i = 0; i < TAMANHO_HASHTABLE; i++) {
878 bool limpo = true;
879 for (int j = 0; j < tamanhoCodificado; j++)
880 if (elementosHT[i][j] != 0) {
881 limpo = false;
882 elementosHT[i][j] = 0;
883 }
884 if (!limpo)
885 usado++;
886 }
887 // reportar estatísticas se existir muito reuso
888 if (usado > 0 && usado * 2 <= colocadosHT) {
889 printf("\nHT: utilização %d%%, reuso: %.2f vezes",
890 usado * 100 / TAMANHO_HASHTABLE,
891 1.0 * colocadosHT / usado);
892 }
893 }
894 else
895 for (int i = 0; i < TAMANHO_HASHTABLE; i++)
896 for (int j = 0; j < tamanhoCodificado; j++)
897 elementosHT[i][j] = 0;
898 colocadosHT = 0;
899 // coloca o estado atual na hasttable, para não ser gerado
900 ExisteHT();
901 }

904bool TProcuraConstrutiva::ExisteHT() {
905 // caso não exista, retorna falso e insere
906 // se existe retorna verdade, e estiver lá outro elemento, substitui
907 unsigned int original = Hash();
908 unsigned int indice = original % TAMANHO_HASHTABLE;
909 for (int i = 0; i < tamanhoCodificado; i++)
910 if (elementosHT[indice][i] != estadoCodHT[i]) {
911 SubstituirHT(indice);
912 colocadosHT++;
913 return false; // não existia
914 }
915 // elemento é igual, mas se o custo for mais alto, não conta
916 // já que este elemento com custo mais baixo, pode conduzir à solução ou melhores soluções
917 // neste caso substitui e retorna que o estado não existe
918 if (custoHT[indice] > custo) {
919 SubstituirHT(indice);
920 return false; // existe mas com um custo mais alto
921 }
922 return true; // igual, estado já analisado
923}
