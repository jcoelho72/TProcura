95// retorna o valor do estado actual, apos procura de profundidade nivel
96int TProcuraAdversa::MiniMax(int nivel)
97{
98 bool noFolha = (nivel == 1 || Parar());
99
100 if (nivel == 0)
101 return MetodoIterativo(false);
102
103 DebugChamada(noFolha);
104 // no final da árvore, retornar (valor da heuristica)
105 if (noFolha) {
106 completo = false; // árvore cortada, a procura deixa de ser completa
107 return NoFolha(true);
108 }
109
110 // expandir o estado
111 TVector<TNo> sucessores;
112 Sucessores(sucessores);
113 // caso não existam sucessores, é como se fosse um nó folha
114 if (sucessores.Empty())
115 return NoFolha(false);
116
117 TVector<int> id; // índice para ordenar os sucessores por heurística
118 OrdenarSucessores(sucessores, id, nivel);
119
120 int resultado = 0, melhor = -1;
121 // processar os sucessores
122 for (int i = 0; i < id.Count(); i++) {
123 DebugExpansao(i, id.Count(), minimizar);
124 int valor;
125 TValorEstado valorConhecido;
126
127 if (((TProcuraAdversa*)sucessores[id[i]])->ValorEstado(valorConhecido, LER) &&
128 valorConhecido.nivel >= nivel)
129 {
130 valor = valorConhecido.valor;
131 if (Parametro(NIVEL_DEBUG) >= PASSOS) {
132 ((TProcuraAdversa*)sucessores[id[i]])->DebugChamada(false);
133 DebugFolha(false, "%-2s%d", Icon(EIcon::MEMORIA), valor);
134 }
135 }
136 else {
137 // chamada recursiva, com um nível a menos, idêntico à procura em profundidade
138 valor = ((TProcuraAdversa*)sucessores[id[i]])->MiniMax(nivel - 1);
139 valorConhecido = { valor, nivel, EXATO };
140 ((TProcuraAdversa*)sucessores[id[i]])->ValorEstado(valorConhecido, GRAVAR);
141 }
142
143 // pretende-se obter o melhor resultado, que oscila entre minimizar ou maximizar
144 if (i == 0 || (minimizar ? resultado > valor : resultado < valor)) {
145 resultado = valor;
146 melhor = id[i];
147 if (nivel > 1 && Parametro(NIVEL_DEBUG) >= PASSOS) { // colocar valor actual alterado
148 ramo.Last() = (minimizar ? RAMO_ESTADO : RAMO_ESTADO2);
149 NovaLinha();
150 printf(" %d", resultado);
151 }
152 // caso de vitória/derrota
153 if (minimizar ? resultado <= custo + 1 - infinito : resultado >= infinito - custo - 1) {
154 DebugFolha(true, " %-2s%d", Icon(resultado < 0 ? EIcon::VIT_PRETA : EIcon::VIT_BRANCA), custo);
155 // listar os nós não explorados
156 if (Parametro(NIVEL_DEBUG) >= PASSOS) {
157 TVector<int> valores;
158 for (int j = i + 1; j < id.Count(); j++)
159 valores += sucessores[id[j]]->debugID;
160 MostraConjunto(valores, Icon(EIcon::ID));
161 }
162 break; // nao e possivel melhorar
163 }
164 }
165 }
166 // todos os sucessores analizados, se houver uma solução melhor, retornar
167 if (melhor >= 0) {
168 if (solucao != NULL)
169 delete solucao;
170 solucao = sucessores[melhor];
171 }
172 DebugCorte(sucessores.Count(), minimizar);
173 LibertarVector(sucessores, melhor);
174 return resultado;

197 TVector<TNo>& sucessores, TVector<int>& id, int nivel)
198{
199 if (Parametro(PODA_CEGA) > 0 &&
200 sucessores.Count() > Parametro(PODA_CEGA))

202 // podar de forma aleatória, nem heurística deve ser calculada
203 while (sucessores.Count() > Parametro(PODA_CEGA)) {
204 int id = TRand::rand() % sucessores.Count();
205 delete sucessores[id];
206 sucessores[id] = sucessores.Last();
207 sucessores.Pop();

216 id.Invert();
217
218 // podar sucessores
219 if (Parametro(PODA_HEURISTICA) > 0 &&
220 Parametro(PODA_HEURISTICA) < id.Count()) {
221 TVector<TNo> manterSuc;
222 for (int i = 0; i < Parametro(PODA_HEURISTICA); i++)
223 manterSuc += sucessores[id[i]];
224 for (int i = Parametro(PODA_HEURISTICA); i < id.Count(); i++)
225 delete sucessores[id[i]];
226 sucessores = manterSuc;
227 id = {};
228 for (int i = 0; i < sucessores.Count(); i++)
229 id += i;
230 }
231
232 }
233 else
234 for (int i = 0; i < sucessores.Count(); i++)
235 id += i;

238// iteração, aumentando o nível progressivamente
239int TProcuraAdversa::MetodoIterativo(int alfaBeta) {
240 int resultado = 0, resOK = 0, nivel = 0;
241 nivelOK = 0;
242 TNo solOK = NULL;
243 do {
244 DebugIteracao(nivel + 1, Icon(EIcon::LIMITE));
245 completo = true;
246 // chamar a profundidade nível 1, e se não resolver, o nível 2, e assim sucessivamente
247 resultado = (alfaBeta ? MiniMaxAlfaBeta(++nivel) : MiniMax(++nivel));
248 if (!Parar() || nivel == 0) {
249 // guardar a última solução realizada sem interrupções, bem como o resultado
250 if (solOK != NULL)
251 delete solOK;
252 solOK = solucao;
253 solucao = NULL;
254 resOK = resultado;
255 nivelOK = nivel;
256 if (Parametro(NIVEL_DEBUG) > NADA && solOK != NULL)
257 printf("\n │ %-2s%-2s %d %-2s%s %-2s%d ",
258 Icon(EIcon::ARVORE), Icon(EIcon::LIMITE), nivel,
259 Icon(EIcon::ACCAO), *Acao(solOK), Icon(EIcon::SUCESSO), resultado);
260 }
261 else
262 completo = false;
263 } while (!completo && !Parar());
264 if (solucao != NULL)
265 delete solucao;
266 solucao = solOK;
267 return resOK;
268}

270// Utilitário para calculo de uma heurística standard em jogos simples (tipicamente sem empates):
271// - calcular o número de ameaças de vitória, para cada lado, de menor comprimento:
272// - qMin - vetor com número de ameaças (1 ou mais) a 1 jogada (na primeira posição), a 2 (na segunda posição), e assim sucessivamente;
273// - qMax - vetor com número de ameaças (1 ou mais) a 1 jogada (na primeira posição), a 2 (na segunda posição), e assim sucessivamente;
274int TProcuraAdversa::MaiorAmeaca(TVector<int>& qMin, TVector<int>& qMax, int maxAmeaca) const
275{
276 double pontos = 0;
277 double base = Parametro(HEUR_BASE) / 100.0;
278 double peso = 1;
279
280 // verificar situações de ganho imediato
281 if (!minimizar && qMax.First() > 0) {
282 return infinito; // Vitória imediata para o max
283 }
284 if (minimizar && qMin.First() > 0) {
285 return -infinito; // Vitória imediata para o min
286 }
287
288 // Ameaças iguais a maxAmeaca ou superior, valem 1, todas as outras valem conforme
289 peso = 1;
290 for (int i = qMin.Count() - 1; i >= 0; i--) {
291 pontos -= qMin[i] * peso;
292 if (i < maxAmeaca) // peço começa a aumentar
293 peso *= base;
294 }
295 peso = 1;
296 for (int i = qMax.Count() - 1; i >= 0; i--) {
297 pontos += qMax[i] * peso;
298 if (i < maxAmeaca) // peço começa a aumentar
299 peso *= base;
300 }
301
302 return (int)(0.5 + infinito * (2.0 / (1.0 + exp(-pontos * 2.0 / infinito)) - 1.0));
303}

305// fim da procura, por corte de nível (ou não haver sucessores), retornar heurística
306int TProcuraAdversa::NoFolha(bool nivel) {
307 int resultado = Heuristica();
308
309 if (resultado >= infinito) {
310 resultado = infinito - custo; // subtrair
311 // jogo ganho pelo branco
312 // se o agente minimiza, tentar o jogo mais longo possível
313 // caso contrário, quanto maior o jogo pior, quer o jogo mais curto
314 // a minimizar, entre um nível 10 e 20, irá preferir o 20, já que -20 é menor que -10
315 // a maximizar, entre um nível 10 e 20, irá preferir o 10, já que 10 é menor que 20
316 }
317 else if (resultado <= -infinito) {
318 resultado = custo - infinito; // somar
319 // jogo ganho pelo preto
320 // se o agente maximiza, tentar o jogo mais longo possível
321 // caso contrário, quanto maior o jogo pior, quer o jogo mais curto
322 // a maximizar, entre 10 e 20, irá preferir 20, sempre é maior
323 // a minimizar, entre 10 e 20, irá preferir 10 que é menor
324 }
325 DebugFolha(false, "%-2s%d", Icon(EIcon::FOLHA), resultado);
326 return resultado;
327}

423// verifica se há um corte alfa/beta, atualizando alfa e beta
424bool TProcuraAdversa::CorteAlfaBeta(int valor, int& alfa, int& beta) {
425 if (minimizar) { // pretas
426 // ver se ja e maximo
427 if (valor <= custo + 1 - infinito) {
428 DebugFolha(true, "%-2s%d", Icon(EIcon::VIT_PRETA), custo);
429 return true;
430 }
431 if (alfa >= valor) {
432 // corte alfa
433 DebugFolha(true, "%-2sα(%d)", Icon(EIcon::CORTE), alfa);
434 return true; // as brancas tem uma alternativa, e escusado continuar a procurar aqui
435 }
436 // atualização beta
437 if (beta > valor) {
438 beta = valor;
439 Debug(PASSOS, false, " → β");
440 }
441 }
442 else { // brancas
443 // ver se atingiu o maximo
444 if (valor >= infinito - custo - 1) {
445 DebugFolha(true, "%-2s%d", Icon(EIcon::VIT_BRANCA), custo);
446 return true;
447 }
448 if (beta <= valor) {
449 // corte beta
450 DebugFolha(true, "%-2sβ(%d)", Icon(EIcon::CORTE), beta);
451 return true; // as pretas tem uma alternativa, e escusado continuar a procurar aqui
452 }
453 // atualização alfa
454 if (alfa < valor) {
455 alfa = valor;
456 Debug(PASSOS, false, " → α");
457 }
458 }
459 // não há corte
460 return false;
461}

554 // jogar ora de brancas ora de pretas, até o jogo terminar
555 while (!SolucaoCompleta()) {
556 ConfiguracaoAtual(configuracoes[njogada % 2 == 0 ? brancas : pretas], GRAVAR);
557 TRand::srand(Parametro(SEMENTE));
558 LimparEstatisticas();
559 int backupDebug = Parametro(NIVEL_DEBUG);
560 Parametro(NIVEL_DEBUG) = NADA;
561 resultado = ExecutaAlgoritmo();
562 Parametro(NIVEL_DEBUG) = backupDebug;
563 if (njogada % 2 == 0)
564 resultados.Last().tempoBrancas += Cronometro(CONT_ALGORITMO);
565 else
566 resultados.Last().tempoPretas += Cronometro(CONT_ALGORITMO);
567
568 if (solucao != NULL) { // efetuado um lance
569 const char* strAcao = Acao(solucao);
570 Copiar(solucao);
571 if (Parametro(NIVEL_DEBUG) >= COMPLETO)
572 printf(" %s", strAcao);
573 njogada++;
574 if (njogada % 2 == 1) // jogada de brancas, colocar o número de jogada (njogada é meia jogada)
575 (lance = "").printf(" %d. %s", (njogada / 2) + 1, strAcao);
576 else // jogada de pretas
577 (lance = "").printf(" %s", strAcao);
578 // concatenar ao registo do jogo
579 resultados.Last().jogo += lance;
580 }
581 else {
582 break; // não há lance efetuado
583 resultado = 0; // erro, mas considerar empatado
584 }
585 }
586 resultados.Last().resultado = Pontos(resultado);
587 resultados.Last().nJogadas = njogada;
588
589 // caso o primeiro a jogar pretenda minimizar, então inverter,
590 // já que o positivo/primeiro a jogar é para as brancas a nível de torneio
591 // mesmo que no jogo o primeiro jogador minimize
592 if ((njogada % 2 == 0) == minimizar)
593 resultados.Last().resultado *= -1;
594
595 // jogo terminou, registar resultado
596 if (torneio != NULL)
597 (*torneio)[brancas][pretas] += resultados.Last().resultado;
598 Debug(COMPLETO, false, " 🏆 %s",
599 (resultados.Last().resultado < 0 ? Icon(EIcon::VIT_PRETA) :
600 (resultados.Last().resultado > 0 ? Icon(EIcon::VIT_BRANCA) :
601 Icon(EIcon::EMPATE))));
602
603 // colocar resultado do jogo no final
604 if (!resultados.Last().jogo.Empty()) {
605 (lance="").printf(" 🏆 %s", (resultados.Last().resultado < 0 ? Icon(EIcon::VIT_PRETA) :
606 (resultados.Last().resultado > 0 ? Icon(EIcon::VIT_BRANCA) :
607 Icon(EIcon::EMPATE))));
608 resultados.Last().jogo += lance;

612void TProcuraAdversa::TesteEmpiricoGestor(TVector<int> instancias, TString ficheiro)
613{
614#ifdef MPI_ATIVO
615 int dados[6] = { 0, 0, 0, 4,0,0 }; // instância, brancas, pretas
616 int64_t dadosD[3];
617 double esperaTrabalhadores = 0, esperaGestor = 0;
618 TVector<double> terminou; // instante em que terminou cada trabalhador
619 TVector<int> trabalhador, trabalhar;
620 TVector<int> atual;
621 double periodoReporte = 60;
622
623 TesteInicio(instancias, atual);
624
625 switch (Parametro(NIVEL_DEBUG)) {
626 case DETALHE: periodoReporte = 10; break;
627 case COMPLETO: periodoReporte = 0; break;
628 }
629 for (int i = 1; i < mpiCount; i++)
630 trabalhador += i;
631
632 terminou.Count(mpiCount);
633 terminou.Reset(0);
634
635 // Ciclo:
636 // 1. Enviar trabalho para os escravos
637 // 2. Encerrar escravos a mais
638 // 3. Receber resultados e repetir 1 ou 2 conforme as necessidades
639
640 TVector<TResultadoJogo> resultados; // guarda as soluções obtidas
641 TVector<TResultadoJogo> tarefas; // guarda informação apenas das tarefas a realizar (sem resultados)
642 Cronometro(CONT_REPORTE, true); // reiniciar cronómetro evento
643
644 // construir todas as tarefas
645 // dois jogadores, brancas é o primeiro a jogar, pretas é o segundo
646 for (int brancas = 0; brancas < configuracoes.Count(); brancas++)
647 for (int pretas = 0; pretas < configuracoes.Count(); pretas++)
648 if (brancas != pretas)
649 for (auto inst : instancias)
650 tarefas += { inst, brancas, pretas };
651
652
653 int totalTarefas = tarefas.Count();
654 Debug(ATIVIDADE, false, "\n ├─ %-2sTarefas:%d %-2sInstâncias: %d %-2sConfigurações: %d %-2sProcessos: %d.",
655 Icon(EIcon::TAREFA), tarefas.Count(),
656 Icon(EIcon::INST), instancias.Count(),
657 Icon(EIcon::CONF), configuracoes.Count(),
658 Icon(EIcon::PROCESSO), trabalhador.Count() + 1) &&
659 fflush(stdout);
660
661 // dar uma tarefa a cada escravo
662 while (!tarefas.Empty() && !trabalhador.Empty()) {
663 auto tarefa = tarefas.Pop();
664 dados[0] = tarefa.instancia;
665 dados[1] = tarefa.brancas;
666 dados[2] = tarefa.pretas;
667 trabalhar += trabalhador.Last();
668 MPI_Send(dados, 3, MPI_INT, trabalhador.Pop(), TAG_TRABALHO, MPI_COMM_WORLD);
669 esperaTrabalhadores += Cronometro(CONT_TESTE); // estava parado até esta altura
670 }
671 // caso existam escravos sem trabalho, mandar fechar todos, não há mais tarefas
672 dados[0] = dados[1] = -1;
673 while (!trabalhador.Empty()) {
674 auto trabalhadorID = trabalhador.Pop();
675 MPI_Send(dados, 3, MPI_INT, trabalhadorID, TAG_TRABALHO, MPI_COMM_WORLD);
676 terminou[trabalhadorID] = Cronometro(CONT_TESTE);
677 }
678
679 // receber resultados e continuar a dar trabalho caso exista
680 while (!trabalhar.Empty()) {
681 MPI_Status stat;
682
683 double inicioEspera = Cronometro(CONT_TESTE);
684 MPI_Recv(dados, 6, MPI_INT, MPI_ANY_SOURCE, TAG_CABECALHO, MPI_COMM_WORLD, &stat);
685 esperaGestor += Cronometro(CONT_TESTE) - inicioEspera;
686 resultados += {dados[0], dados[1], dados[2], dados[3], dados[4]};
687 trabalhar -= stat.MPI_SOURCE;
688 trabalhador += stat.MPI_SOURCE;
689 resultados.Last().jogo.Count(dados[5] + 1);
690 MPI_Recv(resultados.Last().jogo.Data(), dados[5], MPI_CHAR,
691 stat.MPI_SOURCE, TAG_JOGO, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
692 MPI_Recv(dadosD, 3, MPI_LONG_LONG,
693 stat.MPI_SOURCE, TAG_VALORES, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
694 // tempo de brancas e pretas e de espera do trabalhador (em segundos)
695 resultados.Last().tempoBrancas = (double)dadosD[0] / 1000.;
696 resultados.Last().tempoPretas = (double)dadosD[1] / 1000.;
697 esperaTrabalhadores += (double)((int64_t)dadosD[2]) / 1000.;
698
699 if (Parametro(NIVEL_DEBUG) > NADA && Cronometro(CONT_REPORTE) > periodoReporte) {
700 // mostrar uma linha por cada execução
701 Debug(ATIVIDADE, false,
702 "\n ├─ %-2s%-15s %-2s%-5d %-2s%-5d %-2s%-5d %-2s%-5d %-2s%-5d %-2s ",
703 Icon(EIcon::TEMPO), *MostraTempo(Cronometro(CONT_TESTE)),
704 Icon(EIcon::TAREFA), totalTarefas - tarefas.Count(),
705 Icon(EIcon::INST), resultados.Last().instancia,
706 Icon(EIcon::CONF), resultados.Last().brancas,
707 Icon(EIcon::CONF), resultados.Last().pretas,
708 Icon(EIcon::PROCESSO), trabalhador.Last(),
709 Icon(EIcon::IND));
710 fflush(stdout);
711 Cronometro(CONT_REPORTE, true);
712 }
713
714 // ainda há tarefas
715 if (!tarefas.Empty()) {
716 auto tarefa = tarefas.Pop();
717 dados[0] = tarefa.instancia;
718 dados[1] = tarefa.brancas;
719 dados[2] = tarefa.pretas;
720 trabalhar += trabalhador.Last();
721 MPI_Send(dados, 3, MPI_INT, trabalhador.Pop(), TAG_TRABALHO, MPI_COMM_WORLD);
722 }
723 else { // tudo feito, mandar sair
724 dados[0] = dados[1] = -1;
725 auto trabalhadorID = trabalhador.Pop();
726 MPI_Send(dados, 3, MPI_INT, trabalhadorID, TAG_TRABALHO, MPI_COMM_WORLD);
727 terminou[trabalhadorID] = Cronometro(CONT_TESTE);
728 }
729 }
730
731 // contar a espera dos trabalhadores, após terminarem
732 for (int i = 1; i < mpiCount; i++)
733 esperaTrabalhadores += Cronometro(CONT_TESTE) - terminou[i];
734
735 // escrever o ficheiro de resultados
736 int backupCount = mpiCount;
737 double taxaUtilizacaoT = 1. - (esperaTrabalhadores / (Cronometro(CONT_TESTE) * (mpiCount - 1)));
738 double taxaUtilizacaoG = 1. - (esperaGestor / Cronometro(CONT_TESTE));
739 double taxaUtilizacao = 1. - ((esperaTrabalhadores + esperaGestor) / (Cronometro(CONT_TESTE) * mpiCount));
740 mpiCount = 1; // forçar a escrita do ficheiro apenas neste processo
741 RelatorioCSV(resultados, ficheiro) &&
742 Debug(ATIVIDADE, false,
743 "\n ├─ %-2s Ficheiro %s.csv gravado.\n"
744 " │ %-2s Tempo real: %s",
745 Icon(EIcon::RESULT), *ficheiro,
746 Icon(EIcon::TEMPO), *MostraTempo(Cronometro(CONT_TESTE))) &&
747 Debug(ATIVIDADE, false, "\n │ %-2s CPU total: %s",
748 Icon(EIcon::TEMPO), *MostraTempo(Cronometro(CONT_TESTE) * (backupCount - 1))) &&
749 Debug(ATIVIDADE, false, "\n │ %-2s Espera do gestor: %s",
750 Icon(EIcon::TEMPO), *MostraTempo(esperaGestor)) &&
751 Debug(ATIVIDADE, false, "\n │ %-2s Espera trabalhadores: %s",
752 Icon(EIcon::TEMPO), *MostraTempo(esperaTrabalhadores)) &&
753 Debug(ATIVIDADE, false, "\n │ %-2s Utilização:\n │ - Total: %.1f%%\n │ - Gestor: %.1f%%\n │ - Trabalhadores: %.1f%% ",
754 Icon(EIcon::TAXA), taxaUtilizacao * 100, taxaUtilizacaoG * 100, taxaUtilizacaoT * 100);
755 mpiCount = backupCount;
756
757 TesteFim();
758
759#endif
760}
