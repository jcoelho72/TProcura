401void TProcura::InserirConfiguracoes(char* str, TVector<int>& base) {
402 char* pt;
403 TVector<int> currente, produto;
404 TVector<TVector<int>> valores;
405
406 // processar todos os itens a iniciar em P, obtendo informação entre quais existe x
407 pt = strtok(str, " \n\t\r");
408 while (pt) {
409 if (*pt == 'P') {
410 char* pt2;
411 int param;
412 if (pt2 = strchr(pt + 1, '=')) {
413 *pt2 = 0;
414 param = atoi(pt + 1);
415 if (param > 0 && param <= parametro.Count()) {
416 valores.Count(valores.Count() + 1);
417 valores.Last() = {};
418 valores.Last() += param; // primeiro valor é ID do parâmetro
419 valores.Last() += _TV(pt2 + 1); // valores para o parâmetro tomar
420 if (valores.Last().Count() == 2) {
421 // apenas um elemento, altera a configuração atual
422 // (se fosse para alternar, colocava o valor base masi o valor a alternar)
423 int valor = valores.Last().Last();
424 if (valor >= parametro[param - 1].min &&
425 valor <= parametro[param - 1].max)
426 base[param - 1] = valor;
427 }
428 if (valores.Last().Count() <= 2)
429 valores.Count(valores.Count() - 1);
430 }
431 }
432 }
433 else if (*pt == 'x') {
434 valores.Count(valores.Count() + 1);
435 valores.Last() += 0; // produto externo

437 pt = strtok(NULL, " \n\t\r");

441 for (int i = 0; i < valores.Count(); i++)
442 if (valores[i].First() > 0) { // c.c. é o operador produto externo

589 int dados[3] = { 0, 0, 0 }; // instância, configuração
590 double esperaTrabalhadores = 0, esperaGestor = 0;
591 TVector<double> terminou; // instante em que terminou cada trabalhador
592 TVector<int> trabalhador, trabalhar;
593 double periodoReporte = 60;
594 switch (Parametro(NIVEL_DEBUG)) {
595 case DETALHE: periodoReporte = 10; break;
596 case COMPLETO: periodoReporte = 1; break;
597 }
598 for (int i = 1; i < mpiCount; i++)
599 trabalhador += i;
600
601 terminou.Count(mpiCount);
602 terminou.Reset(0);
603
604 // Ciclo:
605 // 1. Enviar trabalho para os escravos
606 // 2. Encerrar escravos a mais
607 // 3. Receber resultados e repetir 1 ou 2 conforme as necessidades
608
609 TVector<TResultado> resultados; // guarda as soluções obtidas
610 TVector<TResultado> tarefas; // guarda informação apenas das tarefas a realizar (sem resultados)
611 TVector<int> atual;
612 int backupID = instancia.valor;
613 Cronometro(CONT_TESTE, true); // reiniciar cronómetro global
614 Cronometro(CONT_REPORTE, true); // reiniciar cronómetro evento
615 for (auto item : instancias)
616 if (item<instancia.min || item>instancia.max)
617 item = -1;
618 instancias -= (-1);
619 ConfiguracaoAtual(atual, LER);
620 if (configuracoes.Empty()) {
621 // não foram feitas configurações, utilizar a atual
622 configuracoes.Count(1);
623 configuracoes.Last() = atual;
624 }
625
626 // construir todas as tarefas
627 for (int configuracao = 0; configuracao < configuracoes.Count(); configuracao++)
628 for (auto inst : instancias)
629 tarefas += { inst, configuracao };
630
631 Debug(ATIVIDADE, false, "\nTeste com %d tarefas, %d instâncias, %d configurações, utilizando %d processo(s). ",
632 tarefas.Count(), instancias.Count(), configuracoes.Count(),
633 trabalhador.Count() + 1) &&
634 fflush(stdout);
635
636 // dar uma tarefa a cada escravo
637 while (!tarefas.Empty() && !trabalhador.Empty()) {
638 auto tarefa = tarefas.Pop();
639 dados[0] = tarefa.instancia;
640 dados[1] = tarefa.configuracao;
641 trabalhar += trabalhador.Last();
642 MPI_Send(dados, 2, MPI_INT, trabalhador.Pop(), TAG_TRABALHO, MPI_COMM_WORLD);
643 esperaTrabalhadores += Cronometro(CONT_TESTE); // estava parado até esta altura
644 }
645 // caso existam escravos sem trabalho, mandar fechar todos, não há mais tarefas
646 dados[0] = dados[1] = -1;
647 while (!trabalhador.Empty()) {
648 auto trabalhadorID = trabalhador.Pop();
649 MPI_Send(dados, 2, MPI_INT, trabalhadorID, TAG_TRABALHO, MPI_COMM_WORLD);
650 terminou[trabalhadorID] = Cronometro(CONT_TESTE);
651 }
652
653 // receber resultados e continuar a dar trabalho caso exista
654 while (!trabalhar.Empty()) {
655 MPI_Status stat;
656
657 if (Parametro(NIVEL_DEBUG) > NADA && Cronometro(CONT_REPORTE) > periodoReporte) {
658 // mostrar uma linha por cada execução
659 Debug(ATIVIDADE, false, "\n%s Faltam %d tarefas, em curso %d",
660 MostraTempo(Cronometro(CONT_TESTE)), tarefas.Count(), trabalhar.Count()) &&
661 fflush(stdout);
662 Cronometro(CONT_REPORTE, true);
663 }
664
665 double inicioEspera = Cronometro(CONT_TESTE);
666 MPI_Recv(dados, 3, MPI_INT, MPI_ANY_SOURCE, TAG_CABECALHO, MPI_COMM_WORLD, &stat);
667 esperaGestor += Cronometro(CONT_TESTE) - inicioEspera;
668 resultados += {dados[0], dados[1]};
669 resultados.Last().valor.Count(dados[2]);
670 trabalhar -= stat.MPI_SOURCE;
671 trabalhador += stat.MPI_SOURCE;
672 MPI_Recv(resultados.Last().valor.Data(), dados[2], MPI_LONG_LONG,
673 stat.MPI_SOURCE, TAG_VALORES, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
674 // tempo de espera do trabalhador
675 esperaTrabalhadores += (double)((int64_t)resultados.Last().valor.Pop()) / 1000.;
676
677 // ainda há tarefas
678 if (!tarefas.Empty()) {
679 auto tarefa = tarefas.Pop();
680 dados[0] = tarefa.instancia;
681 dados[1] = tarefa.configuracao;
682 trabalhar += trabalhador.Last();
683 MPI_Send(dados, 2, MPI_INT, trabalhador.Pop(), TAG_TRABALHO, MPI_COMM_WORLD);
684 }
685 else { // tudo feito, mandar sair
686 dados[0] = dados[1] = -1;
687 auto trabalhadorID = trabalhador.Pop();
688 MPI_Send(dados, 2, MPI_INT, trabalhadorID, TAG_TRABALHO, MPI_COMM_WORLD);
689 terminou[trabalhadorID] = Cronometro(CONT_TESTE);
690 }
691 }
692
693 // contar a espera dos trabalhadores, após terminarem
694 for (int i = 1; i < mpiCount; i++)
695 esperaTrabalhadores += Cronometro(CONT_TESTE) - terminou[i];
696
697 // escrever o ficheiro de resultados
698 int backupCount = mpiCount;
699 double taxaUtilizacaoT = 1. - (esperaTrabalhadores / (Cronometro(CONT_TESTE) * (mpiCount - 1)));
700 double taxaUtilizacaoG = 1. - (esperaGestor / Cronometro(CONT_TESTE));
701 double taxaUtilizacao = 1. - ((esperaTrabalhadores + esperaGestor) / (Cronometro(CONT_TESTE) * mpiCount));
702 mpiCount = 1; // forçar a escrita do ficheiro apenas neste processo
703 RelatorioCSV(resultados, ficheiro) &&
704 Debug(ATIVIDADE, false,
705 "\nFicheiro %s.csv gravado.\n"
706 "Tempo real: %s",
707 ficheiro, MostraTempo(Cronometro(CONT_TESTE))) &&
708 Debug(ATIVIDADE, false, "\nCPU total: %s",
709 MostraTempo(Cronometro(CONT_TESTE) * (backupCount - 1))) &&
710 Debug(ATIVIDADE, false, "\nEspera do gestor: %s",
711 MostraTempo(esperaGestor)) &&
712 Debug(ATIVIDADE, false, "\nEspera total dos trabalhadores: %s",
713 MostraTempo(esperaTrabalhadores)) &&
714 Debug(ATIVIDADE, false, "\nTaxa de utilização:\n- Total: %.1f%%\n- Gestor: %.1f%%\n- Trabalhadores: %.1f%% ",
715 taxaUtilizacao * 100, taxaUtilizacaoG * 100, taxaUtilizacaoT * 100);
716 mpiCount = backupCount;
717
718#endif
719}

1128int TProcura::MelhorResultado(TResultado base, TResultado alternativa) {
1129 // se não resolvido por ambos, retornar igualdade (assumir código -1 para impossível, -2 para não resolvido, menor é melhor)
1130 if (Registo(base, IND_RESULTADO) == -2 && Registo(alternativa, IND_RESULTADO) == -2)
1131 return 0;
1132 // se igual no custo e o tempo menor que 100, retornar igualdade
1133 if (Registo(base, IND_RESULTADO) == Registo(alternativa, IND_RESULTADO) &&
1134 abs(Registo(base, IND_TEMPO) - Registo(alternativa, IND_TEMPO)) / 100 == 0)
1135 return 0;
1136 // primeiro custo (ou não resolvido, -2)
1137 if ((Registo(base, IND_RESULTADO) == -2 &&
