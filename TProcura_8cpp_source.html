610void TProcura::InserirConfiguracoes(char* str, TVector<int>& base) {
611 char* pt, * contexto;
612 TVector<int> currente, produto;
613 TVector<TVector<int>> valores;
614
615 // processar todos os itens a iniciar em P, obtendo informação entre quais existe x
616 pt = compat::strtok(str, " \n\t\r", &contexto);
617 while (pt) {
618 if (*pt == 'P') {
619 char* pt2;
620 int param;
621 if ((pt2 = strchr(pt + 1, '='))) {
622 *pt2 = 0;
623 param = atoi(pt + 1);
624 if (param > 0 && param <= parametro.Count()) {
625 valores.Count(valores.Count() + 1);
626 valores.Last() = {};
627 valores.Last() += param; // primeiro valor é ID do parâmetro
628 valores.Last() += _TV(pt2 + 1); // valores para o parâmetro tomar
629 if (valores.Last().Count() == 2) {
630 // apenas um elemento, altera a configuração atual
631 // (se fosse para alternar, colocava o valor base masi o valor a alternar)
632 int valor = valores.Last().Last();
633 if (valor >= parametro[param - 1].min &&
634 valor <= parametro[param - 1].max)
635 base[param - 1] = valor;
636 }
637 if (valores.Last().Count() <= 2)
638 valores.Count(valores.Count() - 1);
639 }
640 }
641 }
642 else if (*pt == 'x') {
643 valores.Count(valores.Count() + 1);
644 valores.Last() += 0; // produto externo
645 }
646 pt = compat::strtok(NULL, " \n\t\r", &contexto);
647 }
648 // inserir configurações de acordo com o pretendido (produto externo, ou apenas à configuração base)
649 produto = {};
650 for (int i = 0; i < valores.Count(); i++)
651 if (valores[i].First() > 0) { // c.c. é o operador produto externo
652 if (i == valores.Count() - 1 || valores[i + 1].First() != 0) { // não há outro produto externo, colocar na configuração atual
653 produto += i;
654 InserirConfiguracoes(base, produto, valores);
655 produto = {};
656 }
657 else
658 produto += i;
659 }

811void TProcura::TesteEmpiricoGestor(TVector<int> instancias, char* ficheiro)
812{
813#ifdef MPI_ATIVO

815 double esperaTrabalhadores = 0, esperaGestor = 0;

828 trabalhador += i;
829
830 terminou.Count(mpiCount);
831 terminou.Reset(0);
832
833 // Ciclo:
834 // 1. Enviar trabalho para os escravos
835 // 2. Encerrar escravos a mais
836 // 3. Receber resultados e repetir 1 ou 2 conforme as necessidades
837
838 TVector<TResultado> resultados; // guarda as soluções obtidas
839 TVector<TResultado> tarefas; // guarda informação apenas das tarefas a realizar (sem resultados)
840 TVector<int> atual;
841 int backupID = instancia.valor;
842 Cronometro(CONT_TESTE, true); // reiniciar cronómetro global
843 Cronometro(CONT_REPORTE, true); // reiniciar cronómetro evento

863 trabalhador.Count() + 1) &&
864 fflush(stdout);

867 while (!tarefas.Empty() && !trabalhador.Empty()) {
868 auto tarefa = tarefas.Pop();
869 dados[0] = tarefa.instancia;
870 dados[1] = tarefa.configuracao;
871 trabalhar += trabalhador.Last();
872 MPI_Send(dados, 2, MPI_INT, trabalhador.Pop(), TAG_TRABALHO, MPI_COMM_WORLD);
873 esperaTrabalhadores += Cronometro(CONT_TESTE); // estava parado até esta altura
874 }
875 // caso existam escravos sem trabalho, mandar fechar todos, não há mais tarefas
876 dados[0] = dados[1] = -1;
877 while (!trabalhador.Empty()) {
878 auto trabalhadorID = trabalhador.Pop();

880 terminou[trabalhadorID] = Cronometro(CONT_TESTE);
881 }
882
883 // receber resultados e continuar a dar trabalho caso exista

886
887 double inicioEspera = Cronometro(CONT_TESTE);
888 MPI_Recv(dados, 3, MPI_INT, MPI_ANY_SOURCE, TAG_CABECALHO, MPI_COMM_WORLD, &stat);

900 // mostrar uma linha por cada execução
901 Debug(ATIVIDADE, false,
902 "\n ├─ ⏱ %-15s 📋 %-5d ↻ %-5d 🛠️ %-5d 🖥️ %-5d ⚖ ",
903 MostraTempo(Cronometro(CONT_TESTE)),
904 totalTarefas - tarefas.Count(),
905 resultados.Last().instancia,
906 resultados.Last().configuracao,
907 trabalhador.Last());

909 printf("%" PRId64 " ", ind);
910 fflush(stdout);
911 Cronometro(CONT_REPORTE, true);
912 }
913
914 // ainda há tarefas
915 if (!tarefas.Empty()) {
916 auto tarefa = tarefas.Pop();
917 dados[0] = tarefa.instancia;
918 dados[1] = tarefa.configuracao;
919 trabalhar += trabalhador.Last();
920 MPI_Send(dados, 2, MPI_INT, trabalhador.Pop(), TAG_TRABALHO, MPI_COMM_WORLD);
921 }
922 else { // tudo feito, mandar sair
923 dados[0] = dados[1] = -1;
924 auto trabalhadorID = trabalhador.Pop();
925 MPI_Send(dados, 2, MPI_INT, trabalhadorID, TAG_TRABALHO, MPI_COMM_WORLD);
926 terminou[trabalhadorID] = Cronometro(CONT_TESTE);
927 }
928 }
929
930 // contar a espera dos trabalhadores, após terminarem
931 for (int i = 1; i < mpiCount; i++)
932 esperaTrabalhadores += Cronometro(CONT_TESTE) - terminou[i];
933
934 // escrever o ficheiro de resultados
935 int backupCount = mpiCount;
936 double taxaUtilizacaoT = 1. - (esperaTrabalhadores / (Cronometro(CONT_TESTE) * (mpiCount - 1)));
937 double taxaUtilizacaoG = 1. - (esperaGestor / Cronometro(CONT_TESTE));
938 double taxaUtilizacao = 1. - ((esperaTrabalhadores + esperaGestor) / (Cronometro(CONT_TESTE) * mpiCount));
939 mpiCount = 1; // forçar a escrita do ficheiro apenas neste processo
940 RelatorioCSV(resultados, ficheiro) &&
941 Debug(ATIVIDADE, false,
942 "\n ├─ 📄 Ficheiro %s.csv gravado.\n"
943 " │ ⏱ Tempo real: %s",
944 ficheiro, MostraTempo(Cronometro(CONT_TESTE))) &&
945 Debug(ATIVIDADE, false, "\n │ ⏱ CPU total: %s",
946 MostraTempo(Cronometro(CONT_TESTE) * (backupCount - 1))) &&
947 Debug(ATIVIDADE, false, "\n │ ⏱ Espera do gestor: %s",
948 MostraTempo(esperaGestor)) &&
949 Debug(ATIVIDADE, false, "\n │ ⏱ Espera trabalhadores: %s",
950 MostraTempo(esperaTrabalhadores)) &&
951 Debug(ATIVIDADE, false, "\n │ 📊 Utilização:\n │ - Total: %.1f%%\n │ - Gestor: %.1f%%\n │ - Trabalhadores: %.1f%% ",
952 taxaUtilizacao * 100, taxaUtilizacaoG * 100, taxaUtilizacaoT * 100);

954
955 printf("\n═╧═ 🏁 Fim do Teste (🖥️ %d ⏱ %s) ═══", mpiID, MostraTempo(Cronometro(CONT_TESTE)));
956 fflush(stdout);
