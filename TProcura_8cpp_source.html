297// A ou A,B,C - único valor ou enumeração
298// A:B ou A:B:C - intervalo A a B, ou com passo C
299TVector<int> TProcura::ExtraiLista(char* str) {
300 TVector<int> lista;
301 char* pt;
302
303 if (!str || *str == '\0')
304 return lista;
305
306 // separar por vírgulas
307 if (pt = strchr(str, ',')) {
308 *pt = 0;
309 lista = ExtraiLista(str);
310 do {
311 str = pt + 1;
312 if (pt = strchr(str, ','))
313 *pt = 0;
314 lista += ExtraiLista(str);
315 } while (pt);
316 }
317 else {
318 // procurar por : (intervalo)
319 if (pt = strchr(str, ':')) {
320 char* pt2;
321 *pt = 0;
322 int A = atoi(str);
323 int C = 1;
324 A = atoi(str);
325 if (pt2 = strchr(pt + 1, ':')) { // A:B:C
326 *pt2 = 0;
327 if ((C = atoi(pt2 + 1)) <= 0)
328 C = 1;
329 } // c.c. A:B
330 int B = atoi(pt + 1);
331 if (A > B) { // ordem não interessa
332 int aux = A;
333 A = B;
334 B = aux;
335 }
336 for (int i = A; i <= B; i += C)
337 lista.Add(i);
338 }
339 else // inteiro apenas
340 lista.Add(atoi(str));
341 }
342 lista.BeASet();
343 return lista;
344}

406void TProcura::InserirConfiguracoes(char* str, TVector<int>& base) {
407 char* pt;
408 TVector<int> currente, produto;
409 TVector<TVector<int>> valores;
410
411 // processar todos os itens a iniciar em P, obtendo informação entre quais existe x
412 pt = strtok(str, " \n\t\r");
413 while (pt != NULL) {
414 if (*pt == 'P') {
415 char* pt2;
416 int param;
417 if (pt2 = strchr(pt + 1, '=')) {
418 *pt2 = 0;
419 param = atoi(pt + 1);
420 if (param > 0 && param <= parametro.Count()) {
421 valores.Count(valores.Count() + 1);
422 valores.Last().Count(0);
423 valores.Last().Add(param); // primeiro valor é ID do parâmetro
424 valores.Last() += ExtraiLista(pt2 + 1); // valores para o parâmetro tomar
425 if (valores.Last().Count() == 2) {
426 // apenas um elemento, altera a configuração atual
427 // (se fosse para alternar, colocava o valor base masi o valor a alternar)
428 int valor = valores.Last().Last();
429 if (valor >= parametro[param - 1].min &&
430 valor <= parametro[param - 1].max)
431 base[param - 1] = valor;
432 }
433 if (valores.Last().Count() <= 2)
434 valores.Count(valores.Count() - 1);
435 }
436 }
437 }
438 else if (*pt == 'x') {
439 valores.Count(valores.Count() + 1);
440 valores.Last().Add(0); // produto externo
441 }
442 pt = strtok(NULL, " \n\t\r");
443 }
444 // inserir configurações de acordo com o pretendido (produto externo, ou apenas à configuração base)
445 produto.Count(0);
446 for (int i = 0; i < valores.Count(); i++)
447 if (valores[i].First() > 0) { // c.c. é o operador produto externo
448 if (i == valores.Count() - 1 || valores[i + 1].First() != 0) { // não há outro produto externo, colocar na configuração atual
449 produto.Add(i);
450 InserirConfiguracoes(base, produto, valores);
451 produto.Count(0);
452 }
453 else
454 produto.Add(i);
455 }
456}

852int TProcura::MelhorResultado(TResultado base, TResultado alternativa) {
853 // se não resolvido por ambos, retornar igualdade (assumir código -1 para impossível, -2 para não resolvido, menor é melhor)
854 if (Registo(base, indResultado) == -2 && Registo(alternativa, indResultado) == -2)
855 return 0;
856 // se igual no custo e o tempo menor que 100, retornar igualdade
857 if (Registo(base, indResultado) == Registo(alternativa, indResultado) &&
858 abs(Registo(base, indTempo) - Registo(alternativa, indTempo)) / 100 == 0)
859 return 0;
860 // primeiro custo (ou não resolvido, -2)
861 if ((Registo(base, indResultado) == -2 &&
862 Registo(alternativa, indResultado) > -2) ||
863 (Registo(alternativa, indResultado) > 0 &&
864 Registo(base, indResultado) > Registo(alternativa, indResultado)))
865 return -1;
866 if ((Registo(base, indResultado) > -2 &&
867 Registo(alternativa, indResultado) == -2) ||
868 (Registo(base, indResultado) > 0 &&
869 Registo(alternativa, indResultado) > Registo(base, indResultado)))
870 return 1;
871 // agora o tempo
872 return Registo(base, indTempo) < Registo(alternativa, indTempo) ? 1 : -1;
873}
