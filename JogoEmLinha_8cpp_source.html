131bool CJogoEmLinha::SolucaoCompleta(void)
132{
133 // Processar linhas e diagonais
134 for (int linha = 0; linha < inst.N; linha++)
135 if (VerLinha(linha, 0, 0, 1) ||
136 VerLinha(linha, 0, 1, 1) ||
137 VerLinha(linha, inst.M - 1, 1, -1))
138 return true;
139
140 // Processar colunas e diagonais
141 for (int coluna = 0; coluna < inst.M; coluna++)
142 if (VerLinha(0, coluna, 1, 0) ||
143 VerLinha(0, coluna, 1, 1) ||
144 VerLinha(0, coluna, 1, -1))
145 return true;
146
147 // verificar se há espaço para mais jogadas
148 for (int i = 0; i < inst.N * inst.M; i++)
149 if (tabuleiro[i] == '.')
150 return false; // podem ser feitas mais jogadas
151 return true; // não há hipótese de mais jogadas
152}

216int CJogoEmLinha::Heuristica()
217{
218 static int index[][2] = { {0,1},{1,0}, {1,1}, {1,-1} };
219 TVector<int> qMin, qMax;
220 int nMax, nMin;
221
222 heuristica = 0;
223
224 if (ExisteHeuritica())
225 return heuristica;
226
227 qMin.Count(inst.K - 1); // contabilizar ameaças a 1, 2 e 3 (até K-1)
228 qMax.Count(inst.K - 1);
229 qMin.Reset(0);
230 qMax.Reset(0);
231 // processar todas as sequências de 4
232 for (int linha = 0; linha < inst.N; linha++) {
233 for (int coluna = 0; coluna < inst.M; coluna++) {
234 for (int dir = 0; dir < 4; dir++) { // 4 direções (horizontal, vertical, duas diagonais)
235 if ((index[dir][0] > 0 ? linha <= inst.N - inst.K : 1) &&
236 (index[dir][1] > 0 ? coluna <= inst.M - inst.K :
237 index[dir][1] < 0 ? coluna >= inst.K - 1 : 1))
238 {
239 nMax = nMin = 0;
240 for (int i = 0; i < inst.K && (nMax == 0 || nMin == 0); i++)
241 if (Casa(linha + i * index[dir][0], coluna + i * index[dir][1]) == 'x')
242 nMin++;
243 else if (Casa(linha + i * index[dir][0], coluna + i * index[dir][1]) == 'o')
244 nMax++;
245 // verificar resultado exato
246 if (nMax == inst.K) {
247 heuristica = infinito;
248 return TProcuraAdversa::Heuristica();
249 }
250 else if (nMin == inst.K) {
251 heuristica = -infinito;
252 return TProcuraAdversa::Heuristica();
253 }
254 // verificar se quem joga tem ameaça a 1, para ganhar na sua vez de jogar
255 if (minimizar ? nMin == inst.K - 1 && nMax == 0 : nMax == inst.K - 1 && nMin == 0)
256 heuristica = (minimizar ? -infinito : infinito); // vitória em 1, é igual a posição terminal
257 // não reteornar de imediato, já que o jogo pode ter sido já ganho pelo adversário
258
259 // registar resultado
260 if (nMax > 0 && nMin == 0)
261 qMax[inst.K - 1 - nMax]++;
262 else if (nMin > 0 && nMax == 0)
263 qMin[inst.K - 1 - nMin]++;
264 }
265 }
266 }
267 }
268 if (heuristica != 0)
269 return TProcuraAdversa::Heuristica();
270
271 heuristica = MaiorAmeaca(qMin, qMax, inst.K - 1);
272 // retornar a soma das ameaças
273 return TProcuraAdversa::Heuristica();
274}
