643template <class Item>
644void TVector<Item>::QuickSort(int start, int end)
645{
646 if (end - start > 32) {
647 // pivôs median-of-three
648 Exch((start + end) >> 1, end - 1);
649 if (v[end - 1] < v[start]) Exch(end - 1, start);
650 if (v[end] < v[start]) Exch(end, start);
651 if (v[end] < v[end - 1]) Exch(end, end - 1);
652
653 start++; end--;
654
655 // partição
656 int i = start - 1, j = end;
657 Item pivot = v[end];
658 for (;;) {
659 while (v[++i] < pivot);
660 while (pivot < v[--j] && j > start);
661 if (i >= j) break;
662 Exch(i, j);
663 }
664 Exch(i, end);
665
666 QuickSort(start - 1, i - 1);
667 QuickSort(i + 1, end + 1);
668 }

704{

1045template <class Item>
1046int TVector<Item>::Distance(TVector<Item>& other, int type)
1047{
1048 int result = 0;
1049 int n1 = Count(), n2 = other.Count();
1050
1051 if (type == 0) {
1052 result = abs(n1 - n2);
1053 int m = Min(n1, n2);
1054 for (int i = 0; i < m; ++i) {
1055 if (v[i] != other[i]) ++result;
1056 }
1057 }
1058 else if (type == 1) {
1059 int m = Min(n1, n2);
1060 for (int i = 0; i < m; ++i) {
1061 result += abs(v[i] - other[i]);
1062 }
1063 }
1064 else if (type == 2) {
1065 result = Max(n1, n2);
1066 for (int i = 0; i + 1 < n1; ++i) {
1067 int j = other.Find(v[i]);
1068 if (j >= 0 && j + 1 < n2 && v[i + 1] == other[j + 1]) {
1069 --result;
1070 }
1071 }
1072 }
1073 else if (type == 3) {
1074 int rows = n1 + 1, cols = n2 + 1;
1075 TVector<int> mtx(rows * cols);
1076 // inicialização da matriz
1077 for (int i = 0; i < rows; ++i) mtx[i] = i;
1078 for (int j = 1; j < cols; ++j) mtx[j * rows] = j;
1079 // cálculo da distância de edição
1080 for (int i = 1; i < rows; ++i) {
1081 for (int j = 1; j < cols; ++j) {
1082 int cost = (v[i - 1] == other[j - 1] ? 0 : 1);
1083 int val = mtx[(i - 1) + rows * (j - 1)] + cost;
1084 val = Min(val, mtx[(i - 1) + rows * j] + 1);
1085 val = Min(val, mtx[i + rows * (j - 1)] + 1);
1086 mtx[i + rows * j] = val;
1087 }
1088 }
1089 result = mtx[n1 + rows * n2];
1090 }
1091
1092 return result;
1093}
