644template <class Item>
645void TVector<Item>::QuickSort(int start, int end)
646{
647 if (end - start > 32) {
648 // pivôs median-of-three
649 Exch((start + end) >> 1, end - 1);
650 if (v[end - 1] < v[start]) Exch(end - 1, start);
651 if (v[end] < v[start]) Exch(end, start);
652 if (v[end] < v[end - 1]) Exch(end, end - 1);
653
654 start++; end--;
655
656 // partição
657 int i = start - 1, j = end;
658 Item pivot = v[end];
659 for (;;) {
660 while (v[++i] < pivot);
661 while (pivot < v[--j] && j > start);
662 if (i >= j) break;
663 Exch(i, j);
664 }
665 Exch(i, end);
666
667 QuickSort(start - 1, i - 1);
668 QuickSort(i + 1, end + 1);
669 }
670}

705{
706 if (end - start < 3) {

708 if (end - start == 1) {
709 if (v[(*idx)[end]] < v[(*idx)[start]])
710 ExchIdx(end, start);

712 else if (end - start == 2) {
713 if (v[(*idx)[end - 1]] < v[(*idx)[start]])
714 ExchIdx(end - 1, start);

718 ExchIdx(end, end - 1);

720 }
721 else {
722 // pivôs median-of-three em idx
723 ExchIdx((start + end) >> 1, end - 1);
724 if (v[(*idx)[end - 1]] < v[(*idx)[start]])
725 ExchIdx(end - 1, start);
726 if (v[(*idx)[end]] < v[(*idx)[start]])
727 ExchIdx(end, start);
728 if (v[(*idx)[end]] < v[(*idx)[end - 1]])

731 start++; end--;
732
733 // partição em idx
734 int i = start - 1, j = end;

1046template <class Item>
1047int TVector<Item>::Distance(TVector<Item>& other, int type)
1048{
1049 int result = 0;
1050 int n1 = Count(), n2 = other.Count();
1051
1052 if (type == 0) {
1053 result = abs(n1 - n2);
1054 int m = Min(n1, n2);
1055 for (int i = 0; i < m; ++i) {
1056 if (v[i] != other[i]) ++result;
1057 }
1058 }
1059 else if (type == 1) {
1060 int m = Min(n1, n2);
1061 for (int i = 0; i < m; ++i) {
1062 result += abs(v[i] - other[i]);
1063 }
1064 }
1065 else if (type == 2) {
1066 result = Max(n1, n2);
1067 for (int i = 0; i + 1 < n1; ++i) {
1068 int j = other.Find(v[i]);
1069 if (j >= 0 && j + 1 < n2 && v[i + 1] == other[j + 1]) {
1070 --result;
1071 }
1072 }
1073 }
1074 else if (type == 3) {
1075 int rows = n1 + 1, cols = n2 + 1;
1076 TVector<int> mtx(rows * cols);
1077 // inicialização da matriz
1078 for (int i = 0; i < rows; ++i) mtx[i] = i;
1079 for (int j = 1; j < cols; ++j) mtx[j * rows] = j;
1080 // cálculo da distância de edição
1081 for (int i = 1; i < rows; ++i) {
1082 for (int j = 1; j < cols; ++j) {
1083 int cost = (v[i - 1] == other[j - 1] ? 0 : 1);
1084 int val = mtx[(i - 1) + rows * (j - 1)] + cost;
1085 val = Min(val, mtx[(i - 1) + rows * j] + 1);
1086 val = Min(val, mtx[i + rows * (j - 1)] + 1);
1087 mtx[i + rows * j] = val;
1088 }
1089 }
1090 result = mtx[n1 + rows * n2];
1091 }
1092
1093 return result;
1094}
